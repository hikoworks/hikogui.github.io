25[[nodiscard]] inline std::filesystem::path executable_file()
26{
27 if (auto path = win32_GetModuleFileName()) {

34[[nodiscard]] inline generator<std::filesystem::path> data_dirs()
35{
36 // "%LOCALAPPDATA%<Application Vendor><Application Name>\"
37 // FOLDERID_LocalAppData has the default path: %LOCALAPPDATA% (%USERPROFILE%\AppData\Local)
38 if (auto path = win32_SHGetKnownFolderPath(FOLDERID_LocalAppData)) {
39 co_yield *path / get_application_vendor() / get_application_name() / "";
40 } else {
41 throw std::system_error{std::error_code{path.error()}};
42 }
43}

45[[nodiscard]] inline generator<std::filesystem::path> log_dirs()
46{

53[[nodiscard]] inline generator<std::filesystem::path> resource_dirs()
54{
55 // Always look at the resource directory where the executable is located.
56 for (auto const& path : executable_dirs()) {
57 co_yield path / "resources" / "";
58 }
59
60 // Also look in the data directory of the application.
61 for (auto const& path : data_dirs()) {
62 co_yield path / "resources" / "";
63 }
64
65 // If the executable of the application is located in the build directory,
66 // then check the source directories for resources.
67 for (auto const& source_dir : source_dirs()) {
68 co_yield source_dir / "resources" / "";
69
70 // Check the in-tree HikoGUI-library build directory.
71 if (auto path = library_cmake_build_dir(); not path.empty()) {
72 co_yield path / "resources" / "";
73 }
74
75 // Check the in-tree HikoGUI-library source directory.
76 if (auto path = library_cmake_source_dir(); not path.empty()) {
77 co_yield path / "resources" / "";
78 }
79
80 for (auto const& library_source_dir : library_source_dirs()) {
81 // Check the HikoGUI source directory.
82 co_yield library_source_dir / "resources" / "";
83
84 // Check the HikoGUI install directory.
85 co_yield library_source_dir / "share" / "hikogui" / "resources" / "";
86 }
87 }
88}

97[[nodiscard]] inline generator<std::filesystem::path> font_dirs()
98{
