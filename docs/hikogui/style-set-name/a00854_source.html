57 [[nodiscard]] static size_t
58 fill_start(std::span<sfloat_rgba16> src, size_t src_nr, unsigned int pass, size_t width, size_t height) noexcept
59 {
60 // clang-format off
61 if (pass == 0 and src.size() == width) {
62 return src_nr;
63 } else switch (pass) {
64 case 0: return src_nr * 8 + 0;
65 case 1: return src_nr * 8 + 0;
66 case 2: return src_nr * 8 + 4;
67 case 3: return src_nr * 4 + 0;
68 case 4: return src_nr * 4 + 2;
69 case 5: return src_nr * 2 + 0;
70 case 6: return src_nr * 2 + 1;
71 default: return height;
72 }
73 // clang-format on
74 }

87 [[nodiscard]] static size_t fill(
88 std::span<sfloat_rgba16> src,
89 size_t src_nr,
90 unsigned int pass,
91 unsigned int step,
92 size_t width,
93 size_t height,
94 std::span<sfloat_rgba16> dst) noexcept
95 {
96 auto const next_step = step + 1;
97
98 if (pass == 0 and src.size() == width) {
99 std::copy(src.begin(), src.end(), dst.begin());
100 return height;
101
102 } else {
103 switch (pass) {
104 case 0: // x=%8+0, y=%8+0
105 for (size_t x = 0; x != width; ++x) {
106 dst[x] = src[x >> 3];
107 }
108 return next_step < 8 ? src_nr * 8 + 0 + next_step : height;
109
110 case 1: // x=%8+4, y=%8+0
111 for (size_t x = 0; x != width; ++x) {
112 if (x % 8 >= 4) {
113 dst[x] = src[(x - 4) >> 3];
114 }
115 }
116 return next_step < 8 ? src_nr * 8 + 0 + next_step : height;
117
118 case 2: // x=%4+0, y=%8+4
119 for (size_t x = 0; x != width; ++x) {
120 dst[x] = src[x >> 2];
121 }
122 return next_step < 4 ? src_nr * 8 + 4 + next_step : height;
123
124 case 3: // x=%4+2, y=%4+0
125 for (size_t x = 0; x != width; ++x) {
126 if (x % 4 >= 2) {
127 dst[x] = src[(x - 2) >> 2];
128 }
129 }
130 return next_step < 4 ? src_nr * 4 + 0 + next_step : height;
131
132 case 4: // x=%2+0, y=%4+2
133 for (size_t x = 0; x != width; ++x) {
134 dst[x] = src[x >> 1];
135 }
136 return next_step < 2 ? src_nr * 4 + 2 + next_step : height;
137
138 case 5: // x=%2+1, y=%2+0
139 for (size_t x = 0; x != width; ++x) {
140 if (x % 2 >= 1) {
141 dst[x] = src[(x - 1) >> 1];
142 }
143 }
144 return next_step < 2 ? src_nr * 2 + 0 + next_step : height;
145
146 case 6: // x=%1+0, y=%2+1
147 std::copy(src.begin(), src.end(), dst.begin());
148 return height;
149
150 default:
151 return height;
152 }
153 }
154 }

379 void info_callback(png_structp png_ptr, png_infop info_ptr) noexcept
380 {
381 _width = png_get_image_width(png_ptr, info_ptr);
382 _height = png_get_image_height(png_ptr, info_ptr);
383 _bit_depth = png_get_bit_depth(png_ptr, info_ptr);
384 _interlace_type = _png_get_interlace_type(png_ptr, info_ptr);
385 _color_type = _png_get_color_type(png_ptr, info_ptr);
386 _color_primaries = _png_get_cHRM(png_ptr, info_ptr);
387 _iccp_profile = _png_get_iCCP(png_ptr, info_ptr);
388
389 // First convert palette and gray images to RGB (with optional alpha)
390 if (_color_type == png_color_type::palette) {
391 png_set_palette_to_rgb(png_ptr);
392
393 } else if (_color_type == png_color_type::gray or _color_type == png_color_type::gray_alpha) {
394 if (_bit_depth < 8) {
395 png_set_expand_gray_1_2_4_to_8(png_ptr);
396 }
397 png_set_gray_to_rgb(png_ptr);
398 }
399
400 // Expand the image to 16-bit per channel.
401 // This is needed to have enough precision for conversion to linear space.
402 if (_bit_depth < 16) {
403 png_set_expand_16(png_ptr);
404 }
405
406 // If the image is missing an alpha channel, then add one.
407 if (png_get_valid(png_ptr, info_ptr, PNG_INFO_tRNS)) {
408 png_set_tRNS_to_alpha(png_ptr);
409 } else if (_color_type == png_color_type::rgb) {
410 png_set_add_alpha(png_ptr, 0xffff, PNG_FILLER_AFTER);
411 }
412
413 constexpr auto use_premultiplied_alpha = true;
414 if constexpr (use_premultiplied_alpha) {
415 // PNG_ALPHA_STANDARD: The RGB values are premultiplied with the
416 // alpha value. The RGB values are always
417 // linearly encoded, even if the specified gamma
418 // is not PNG_GAMMA_LINEAR. Instead the gamma
419 // value specified is the default gamma value
420 // for the file when the file does not include
421 // a gAMA chunk.
422 //
423 // PNG_DEFAULT_sRGB: The default sRGB-like gama value to use when
424 // the file does not have a gAMA chunk.
425 png_set_alpha_mode(png_ptr, PNG_ALPHA_STANDARD, PNG_DEFAULT_sRGB);
426
427 } else {
428 // PNG_ALPHA_PNG: The RGB values are not premultiplied with the alpha value.
429 // But the RGB values are gamma encoded.
430 //
431 // PNG_DEFAULT_sRGB: First set the default "gamma" value to sRGB in case
432 // the file does not have a gAMA chunk.
433 png_set_alpha_mode(png_ptr, PNG_ALPHA_PNG, PNG_DEFAULT_sRGB);
434 // PNG_GAMMA_LINEAR: Convert RGB values to linear space.
435 png_set_alpha_mode(png_ptr, PNG_ALPHA_PNG, PNG_GAMMA_LINEAR);
436 }
437
438 // Calculate the optional color conversion matrix.
439 if (_color_primaries) {
440 auto const image_to_XYZ = color_primaries_to_RGBtoXYZ(*_color_primaries);
441 _color_matrix = XYZ_to_sRGB * image_to_XYZ;
442 } else {
443 _color_matrix = std::nullopt;
444 }
445
446 // Calculate how to convert 16-bit color components to float.
447 switch (_iccp_profile) {
448 case png_iccp_profile::unknown:
449 _component_scale = 1.0f / 65535.0f;
450 break;
451 case png_iccp_profile::ITUR_2100_PQ_FULL:
452 // HDR full scale luminosity is 1000 cd/m2.
453 _component_scale = 12.5f / 65535.0f;
454 break;
455 }
456
457 if (_delegate == nullptr) {
458 png_error(png_ptr, "PNG: info_callback() delegate is null");
459 }
460 try {
461 _delegate->info(this);
462 } catch (...) {
463 png_error(png_ptr, "PNG: info_callback() delegate->info() failed");
464 }
465
466 // Check if the rowbytes is correct: RRGGBBAA, 8 bytes.
467 png_read_update_info(png_ptr, info_ptr);
468 if (png_get_rowbytes(png_ptr, info_ptr) != _width * 8) {
469 png_error(png_ptr, "PNG: info_callback() rowbytes != width * 8");
470 }
471
472 // Reserve the row to pass to the delegate later on.
473 _row.resize(_width);
474 }

476 void row_callback(png_structp png_ptr, png_bytep src, png_uint_32 src_nr, int pass)
477 {
478 auto const num_columns = [&]{
479 switch (_interlace_type) {
480 case png_interlace_type::none:
481 return width();
482 case png_interlace_type::adam7:
483 return width() >> PNG_PASS_COL_SHIFT(pass);
484 }
485 std::unreachable();
486 }();
487
488 auto const float_src = std::span{_row.data(), num_columns};
489
490 auto const c_scale = _component_scale;
491 constexpr auto a_scale = 1.0f / 65535.0f;
492
493 if (_color_matrix) {
494 auto const M = *_color_matrix;
495 auto off = size_t{0};
496 for (auto x = size_t{0}; x != num_columns; ++x, off += 8) {
497 auto const RR = (static_cast<uint16_t>(src[off + 0]) << 8) | static_cast<uint16_t>(src[off + 1]);
498 auto const GG = (static_cast<uint16_t>(src[off + 2]) << 8) | static_cast<uint16_t>(src[off + 3]);
499 auto const BB = (static_cast<uint16_t>(src[off + 4]) << 8) | static_cast<uint16_t>(src[off + 5]);
500 auto const AA = (static_cast<uint16_t>(src[off + 6]) << 8) | static_cast<uint16_t>(src[off + 7]);
501
502 float_src[x] = M * color{RR * c_scale, GG * c_scale, BB * c_scale, AA * a_scale};
503 }
504
505 } else {
506 auto off = size_t{0};
507 for (auto x = size_t{0}; x != num_columns; ++x, off += 8) {
508 auto const RR = (static_cast<uint16_t>(src[off + 0]) << 8) | static_cast<uint16_t>(src[off + 1]);
509 auto const GG = (static_cast<uint16_t>(src[off + 2]) << 8) | static_cast<uint16_t>(src[off + 3]);
510 auto const BB = (static_cast<uint16_t>(src[off + 4]) << 8) | static_cast<uint16_t>(src[off + 5]);
511 auto const AA = (static_cast<uint16_t>(src[off + 6]) << 8) | static_cast<uint16_t>(src[off + 7]);
512
513 float_src[x] = color{RR * c_scale, GG * c_scale, BB * c_scale, AA * a_scale};
514 }
515 }
516
517 if (_delegate == nullptr) {
518 png_error(png_ptr, "PNG: row_callback() delegate is null");
519 }
520 try {
521 _delegate->row(this, float_src, src_nr, pass);
522 } catch (...) {
523 png_error(png_ptr, "PNG: row_callback() delegate->row() failed");
524 }
525 }

568 [[noreturn]] static void _user_error_fn(png_structp png_ptr, png_const_charp msg) noexcept
569 {
570 auto const self = std::launder(static_cast<png_loader*>(png_get_error_ptr(png_ptr)));
571 assert(self != nullptr);
572
573 self->_last_error_message = msg;
574 std::longjmp(png_jmpbuf(png_ptr), 1);
575 }
576
577 void _cleanup() noexcept
578 {
579 _delegate = nullptr;
580 if (std::setjmp(png_jmpbuf(_png_ptr))) {
581 // At this point if something goes wrong, it must be really bad.
582 std::abort();
583 }
584
585 if (_info_ptr) {
586 png_destroy_info_struct(_png_ptr, &_info_ptr);
587 }
588
589 if (_png_ptr) {
590 png_destroy_read_struct(&_png_ptr, nullptr, nullptr);
591 }
592 }
