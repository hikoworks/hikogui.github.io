81template<std::random_access_iterator It, typename UnaryPredicate>
82constexpr It rfind_if(It first, It last, UnaryPredicate predicate)
83{
84 if (first == last) {
85 return last;
86 }
87
88 auto i = last;
89 do {
90 --i;
91 if (predicate(*i)) {
92 return i;
93 }
94 } while (i != first);
95
96 // Not found.
97 return last;
98}

158template<typename ConstIt, typename It, typename UnaryPredicate>
159constexpr It rfind_cluster(ConstIt first, It start, UnaryPredicate predicate)
160{
161 auto const cluster_id = predicate(*start);
162
163 if (start == first) {
164 return first;
165 }
166
167 auto i = start - 1;
168 while (true) {
169 if (predicate(*i) != cluster_id) {
170 return (i + 1);
171 }
172
173 if (i == first) {
174 return i;
175 }
176 --i;
177 }
178 std::unreachable();
179}

199template<typename It, typename S, typename F>
200inline void for_each_cluster(It first, It last, S IsClusterSeperator, F Function)
201{
202 if (first == last) {
203 return;
204 }
205
206 // If the first item is a cluster separator skip over it.
207 if (IsClusterSeperator(*first)) {
208 first++;
209 }
210
211 for (auto i = first; i != last;) {
212 auto j = std::find_if(i, last, IsClusterSeperator);
213 Function(i, j);
214
215 auto skipOverSeperator = (j == last) ? 0 : 1;
216 i = j + skipOverSeperator;
217 }
218}

220template<typename InputIt1, typename InputIt2, typename BinaryPredicate>
221inline std::pair<InputIt1, InputIt2>
222rmismatch(InputIt1 first1, InputIt1 last1, InputIt2 first2, InputIt2 last2, BinaryPredicate predicate) noexcept
223{
224 auto i1 = last1;
225 auto i2 = last2;
226
227 while (true) {
228 if (i1 == first1 && i2 == first2) {
229 return {last1, last2};
230 } else if (i1 == first1) {
231 return {last1, --i2};
232 } else if (i2 == first2) {
233 return {--i1, last2};
234 }
235
236 if (!predicate(*(--i1), *(--i2))) {
237 return {i1, i2};
238 }
239 }
240}

278auto shuffle_by_index(auto first, auto last, auto indices_first, auto indices_last, auto index_op) noexcept
279{
280 std::size_t src_size = std::distance(first, last);
281
282 // Keep track of index locations during shuffling of items.
283 auto src_indices = std::vector<std::size_t>{};
284 src_indices.reserve(src_size);
285 for (std::size_t i = 0; i != src_size; ++i) {
286 src_indices.push_back(i);
287 }
288
289 std::size_t dst = 0;
290 for (auto it = indices_first; it != indices_last; ++it, ++dst) {
291 auto const index = index_op(*it);
292 hi_assert_bounds(index, src_indices);
293
294 auto src = [&src_indices, index]() {
295 auto src = index;
296 do {
297 src = src_indices[src];
298 } while (src_indices[src] != index);
299 return src;
300 }();
301
302 if (src != dst) {
303 std::iter_swap(first + src, first + dst);
304 std::iter_swap(begin(src_indices) + src, begin(src_indices) + dst);
305 }
306 }
307
308 return first + dst;
309}

341{
342 for (auto it = data_first; it != data_last; ++it) {

360template<typename DataIt, typename ValueIt>
361DataIt back_strip(DataIt data_first, DataIt data_last, ValueIt value_first, ValueIt value_last) noexcept
362{
363 auto it = data_last;
364 while (it != data_first) {
365 if (std::find(value_first, value_last, *(--it)) == value_last) {
366 // Return an iterator one beyond the data that is not part of the value set.
367 return it + 1;
368 }
369 }
370
371 return data_first;

383template<std::endian Endian = std::endian::native, typename T, std::unsigned_integral Key>
384[[nodiscard]] constexpr T *fast_lower_bound(std::span<T> table, Key const& key) noexcept
385{
386 auto base = table.data();
387 auto len = table.size();
388
389 // A faster lower-bound search with less branches that are more predictable.
390 while (len > 1) {
391 hi_axiom_not_null(base);
392
393 auto const half = len / 2;
394
395 if (load<Key, Endian>(base + half - 1) < key) {
396 base += half;
397 }
398 len -= half;
399 }
400 if (load<Key, Endian>(base) < key) {
401 return nullptr;
402 }
403 return base;
404}

408template<std::endian Endian = std::endian::native, typename T, std::unsigned_integral Key>
409[[nodiscard]] constexpr T *fast_binary_search_eq(std::span<T> table, Key const& key) noexcept
410{
411 auto const ptr = fast_lower_bound<Endian>(table, key);
412 if (ptr == nullptr) {

437template<
438 std::forward_iterator InputIt,
439 std::sentinel_for<InputIt> InputLast,
440 typename OutputIt,
441 std::invocable<typename std::iterator_traits<InputIt>::value_type&> Predicate>
442InputIt remove_transform_if(InputIt first, InputLast last, OutputIt d_first, Predicate const& predicate)
443{
444 auto result = first;
445 for (; first != last; ++first) {
446 if (auto transformed_item = predicate(*first)) {
447 *d_first++ = std::move(*transformed_item);
448 } else if (result == first) {
449 ++result;
450 } else {
451 *result++ = std::move(*first);
452 }
453 }
454 return result;
455}
