93 [[nodiscard]] font_glyph_ids find_glyphs(grapheme g) const
94 {
95 // Create a glyph_ids object for a single grapheme.
96 auto r = font_glyph_ids{};
97
98 // First try composed normalization
99 for (auto const c : g.composed()) {
100 if (auto const glyph_id = find_glyph(c)) {
101 r.push_back(glyph_id);
102 } else {
103 r.clear();
104 return r;
105 }
106 }
107
108 // Now try decomposed normalization
109 for (auto const c : g.decomposed()) {
110 if (auto const glyph_id = find_glyph(c)) {
111 r.push_back(glyph_id);
112 } else {
113 r.clear();
114 return r;
115 }
116 }
117
118 return r;
119 }

133 bool find_glyphs(std::span<grapheme const> graphemes, std::span<font_glyph_ids> r) const
134 {
135 assert(graphemes.size() == r.size());
136 assert(not id.empty());
137
138 for (auto i = size_t{0}; i != graphemes.size(); ++i) {
139 auto const& g = graphemes[i];
140
141 r[i] = find_glyphs(g);
142 if (r[i].empty()) {
143 return false;
144 }
145 }
146
147 return true;
148 }
