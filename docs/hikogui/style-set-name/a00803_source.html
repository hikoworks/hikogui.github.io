54 template<typename Func, typename... Args>
55 std::future<std::invoke_result_t<Func, Args...>> async(Func&& func, Args&&... args)
56 {
57 using invoke_result = std::invoke_result_t<Func, Args...>;
58
59 // std::function can not be used with move-only types, so we use a
60 // shared_ptr to a promise instead.
61 auto promise = std::make_shared<std::promise<invoke_result>>();
62 auto future = promise->get_future();
63
64 {
65 auto const _ = std::scoped_lock(_mutex);
66 _fifo.push_back(
67 [promise = std::move(promise), func = std::forward<Func>(func), ... args = std::forward<Args>(args)]() mutable {
68 if constexpr (std::is_void_v<invoke_result>) {
69 func(args...);
70 promise->set_value();
71 } else {
72 promise->set_value(func(args...));
73 }
74 });
75 }
76 _condition.notify_one();
77
78 initialize_thread_pool();
79 return future;
80 }
81
82private:
83 mutable std::mutex _mutex;
84 mutable std::condition_variable _condition;
85 fifo<std::function<void()>> _fifo;
86 std::vector<std::jthread> _threads;
87 std::stop_source _stop_source;
88
89 void worker_thread(std::stop_token stop_token, size_t worker_index)
90 {
91 set_thread_name(std::format("async_worker{}", worker_index));
92
93 auto lock = std::unique_lock(_mutex);
94 while (not _fifo.empty() or not stop_token.stop_requested()) {
95 _condition.wait(lock, [&] {
96 return not _fifo.empty() or stop_token.stop_requested();
97 });
98
99 if (not _fifo.empty()) {
100 auto task = std::move(_fifo.front());
101 _fifo.pop_front();
102 lock.unlock();
103
104 // Don't hold the lock while we are executing the task.
105 // This allows other worker threads to pick up others tasks.
106 task();
107
108 lock.lock();
109 }
110 }
111 }

113 void initialize_thread_pool()
114 {
115 static auto const num_concurrent_threads = [] {
116 size_t r = std::thread::hardware_concurrency();
117 if (r == 0) {
118 r = 1;
119 } else if (r > 1) {
120 r -= 1;
121 }
122 return r;
123 }();
124
125 auto const _ = std::scoped_lock(_mutex);
126 auto const num_tasks = _fifo.size();
127 auto const num_concurrent_tasks = std::min(num_tasks, num_concurrent_threads);
128 if (num_concurrent_tasks <= _threads.size()) {
129 return;
130 }
131
132 _threads.emplace_back(
133 [this](std::stop_token stop_token, size_t worker_index) {
134 this->worker_thread(stop_token, worker_index);
135 },
136 _stop_source.get_token(),
137 _threads.size());
138 }
139};
