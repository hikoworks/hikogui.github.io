39template<otype_GDEF_table table>
40[[nodiscard]] inline std::span<std::byte const> otype_GDEF_get_table(std::span<std::byte const> bytes)
41{
42 struct header_type {
43 big_uint16_buf_t major_version;
44 big_uint16_buf_t minor_version;
45 big_uint16_buf_t glyph_class_def_offset;
46 big_uint16_buf_t attach_list_offset;
47 big_uint16_buf_t lig_caret_list_offset;
48 big_uint16_buf_t mark_attach_class_def_offset;
49 };
50
51 auto offset = size_t{0};
52 auto const header = implicit_cast<header_type>(offset, bytes);
53
54 if (*header.major_version != 1) {
55 throw otype_file_error("GDEF table version not supported");
56 }
57
58 if constexpr (table == otype_GDEF_table::glyph_class_def) {
59 return bytes.subspan(gsl::narrow_cast<size_t>(*header.glyph_class_def_offset) * 2);
60
61 } else if constexpr (table == otype_GDEF_table::attach_list) {
62 return bytes.subspan(gsl::narrow_cast<size_t>(*header.attach_list_offset) * 2);
63
64 } else if constexpr (table == otype_GDEF_table::lig_caret_list) {
65 return bytes.subspan(gsl::narrow_cast<size_t>(*header.lig_caret_list_offset) * 2);
66
67 } else if constexpr (table == otype_GDEF_table::mark_attach_class_def) {
68 return bytes.subspan(gsl::narrow_cast<size_t>(*header.mark_attach_class_def_offset) * 2);
69
70 } else if constexpr (table == otype_GDEF_table::mark_glyph_set) {
71 if (*header.minor_version < 2) {
72 throw otype_file_error("GDEF table does not support mark_glyph_set_offset");
73 }
74 auto const mark_glyph_set_offset = implicit_cast<big_uint16_buf_t>(offset, bytes);
75 return bytes.subspan(gsl::narrow_cast<size_t>(*mark_glyph_set_offset) * 2);
76
77 } else {
78 static_assert(false, "Unknown GDEF table");
79 }
80}

88[[nodiscard]] inline std::variant<int16_t,uint16_t> otype_GDEF_caret_value(std::span<std::byte const> bytes)
89{
90 struct caret_value_type1 {
91 big_uint16_buf_t format;
92 big_int16_buf_t coordinate;
93 };
94
95 struct caret_value_type2 {
96 big_uint16_buf_t format;
97 big_uint16_buf_t caret_value_point_index;
98 };
99
100 struct caret_value_type3 {
101 big_uint16_buf_t format;
102 big_int16_buf_t coordinate;
103 big_uint16_buf_t device_table_offset;
104 };
105
106 auto const caret_value_format = implicit_cast<big_uint16_buf_t>(bytes);
107 switch (*caret_value_format) {
108 case 1:
109 return *implicit_cast<caret_value_type1>(bytes).coordinate;
110
111 case 2:
112 return *implicit_cast<caret_value_type2>(bytes).caret_value_point_index;
113
114 case 3:
115 return *implicit_cast<caret_value_type3>(bytes).coordinate;
116
117 default:
118 throw otype_file_error("caret value format not supported");
119 }
120}

122[[nodiscard]] inline generator<std::variant<int16_t,uint16_t>> otype_GDEF_lig_glyph_search(std::span<std::byte const> bytes)
123{
124 struct lig_glyph_type {
125 big_uint16_buf_t caret_count;
126 };
127
128 auto offset = size_t{0};
129 auto const lig_glyph = implicit_cast<lig_glyph_type>(offset, bytes);
130 auto const caret_value_offsets = implicit_cast<big_uint16_buf_t>(offset, bytes, *lig_glyph.caret_count);
131
132 auto r = lean_vector<std::variant<int16_t,uint16_t>>{};
133 r.reserve(*lig_glyph.caret_count);
134
135 for (auto const caret_value_offset : caret_value_offsets) {
136 auto const caret_value_bytes = bytes.subspan(gsl::narrow_cast<size_t>(*caret_value_offset) * 2);
137 co_yield otype_GDEF_caret_value(caret_value_bytes);
138 }
139}

142[[nodiscard]] inline generator<std::variant<int16_t,uint16_t>> otype_GDEF_lig_caret_list_search(std::span<std::byte const> bytes, glyph_id id)
143{
144 struct header_type {
145 big_uint16_buf_t coverage_offset;
146 big_uint16_buf_t lig_glyph_count;
147 };
148
149 auto offset = size_t{0};
150 auto const header = implicit_cast<header_type>(offset, bytes);
151
152 auto const coverage_bytes = bytes.subspan(gsl::narrow_cast<size_t>(*header.coverage_offset) * 2);
153
154 auto const coverage_index = otype_coverage_search(coverage_bytes, id);
155 if (not coverage_index or *coverage_index >= *header.lig_glyph_count) {
156 co_return;
157 }
158
159 auto const lig_glyph_offsets = implicit_cast<big_uint16_buf_t>(offset, bytes, *header.lig_glyph_count);
160 auto const lig_glyph_offset = *lig_glyph_offsets[*coverage_index];
161 auto const lig_glyph_bytes = bytes.subspan(gsl::narrow_cast<size_t>(lig_glyph_offset) * 2);
162
163 for (auto const caret_value : otype_GDEF_lig_glyph_search(lig_glyph_bytes)) {
164 co_yield caret_value;
165 }
166}

174[[nodiscard]] inline bool otype_GDEF_search_mark_glyph(std::span<std::byte const> bytes, uint16_t mark_filtering_set, glyph_id id)
175{
176 struct mark_glyph_set_type {
177 big_uint16_buf_t format;
178 big_uint16_buf_t mark_glyph_count;
179 };
180
181 auto const mark_glyph_set_bytes = otype_GDEF_get_table<otype_GDEF_table::mark_glyph_set>(bytes);
182
183 auto offset = size_t{0};
184 auto const mark_glyph_set_header = implicit_cast<mark_glyph_set_type>(offset, mark_glyph_set_bytes);
185 if (*mark_glyph_set_header.format != 1) {
186 throw otype_file_error("mark_glyph_set format not supported");
187 }
188
189 auto const mark_glyph_count = *mark_glyph_set_header.mark_glyph_count;
190 auto const mark_glyph_set_offsets = implicit_cast<big_uint32_buf_t>(offset, mark_glyph_set_bytes, mark_glyph_count);
191
192 if (mark_filtering_set >= mark_glyph_count) {
193 throw otype_file_error("mark_filtering_set is out of range");
194 }
195
196 auto const mark_glyph_set_offset = *mark_glyph_set_offsets[mark_filtering_set];
197 auto const mark_glyph_set_coverage_bytes = mark_glyph_set_bytes.subspan(mark_glyph_set_offset);
198 return static_cast<bool>(otype_coverage_search(mark_glyph_set_coverage_bytes, id));
199}

248[[nodiscard]] inline bool
249otype_GDEF_filter_glyph(std::span<std::byte const> bytes, glyph_id id, size_t index, uint16_t lookup_flag, uint16_t mark_filtering_set)
250{
251 auto const lookup_flag_ = otype_lookup_flag{lookup_flag};
252
253 auto const glyph_class = [&]() {
254 if (to_bool(lookup_flag_ & otype_lookup_flag::ignore_ligatures)) {
255 return otype_GDEF_search_class(bytes, id);
256 } else {
257 return index == 0 ? otype_GDEF_glyph_class::base : otype_GDEF_glyph_class::mark;
258 }
259 }();
260
261 if (to_bool(lookup_flag_ & otype_lookup_flag::ignore_base_glyphs) and glyph_class == otype_GDEF_glyph_class::base) {
262 return true;
263 }
264 if (to_bool(lookup_flag_ & otype_lookup_flag::ignore_marks) and glyph_class == otype_GDEF_glyph_class::mark) {
265 return true;
266 }
267 if (to_bool(lookup_flag_ & otype_lookup_flag::ignore_ligatures) and glyph_class == otype_GDEF_glyph_class::ligature) {
268 return true;
269 }
270 if (to_bool(lookup_flag_ & otype_lookup_flag::use_mark_filtering_set) and glyph_class == otype_GDEF_glyph_class::mark) {
271 if (not otype_GDEF_search_mark_glyph(bytes, mark_filtering_set, id)) {
272 // Skip all marks that are not in the mark filtering set.
273 return true;
274 }
275 }
276
277 return false;
278}
