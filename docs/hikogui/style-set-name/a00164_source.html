70 [[nodiscard]] constexpr friend constraints merge_row(constraints const& a, constraints const& b) noexcept
71 {
72 auto const new_baseline_offset = [&] {
73 if (a.baseline_priority > b.baseline_priority) {
74 return a.baseline_offset;
75 } else if (a.baseline_priority < b.baseline_priority) {
76 return b.baseline_offset;
77 } else if (a.vertical_alignment == b.vertical_alignment) {
78 switch (a.vertical_alignment) {
79 case vertical_alignment::top:
80 return std::min(a.baseline_offset, b.baseline_offset);
81 case vertical_alignment::middle:
82 return (a.baseline_offset + b.baseline_offset) * 0.5f;
83 case vertical_alignment::bottom:
84 return std::max(a.baseline_offset, b.baseline_offset);
85 }
86 } else {
87 return a.baseline_offset;
88 }
89 }();
90
91 return constraints{
92 extent2{a.size.width() + b.size.width(), std::max(a.size.height(), b.size.height())},
93 a.expand_width | b.expand_width,
94 a.expand_height | b.expand_height,
95 hi::margins{
96 a.margins.left() + b.margins.left(),
97 std::max(a.margins.bottom(), b.margins.bottom()),
98 a.margins.right() + b.margins.right(),
99 std::max(a.margins.top(), b.margins.top()),
100 },
101 new_baseline_offset,
102 max(a.baseline_priority, b.baseline_priority),
103 a.vertical_alignment
104 };
105 }
