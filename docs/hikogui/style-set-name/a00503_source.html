137 [[nodiscard]] constexpr style_specificity specificity() const noexcept
138 {
139 auto b = size_t{};
140 auto c = size_t{};
141 auto d = size_t{};
142
143 for (auto const& segment : *this) {
144 b += static_cast<size_t>(not segment.id.empty());
145 c += static_cast<size_t>(not segment.classes.empty());
146 c += static_cast<size_t>(not segment.pseudo_classes.empty());
147 d += static_cast<size_t>(not segment.name.empty());
148 }
149
150 b = std::min(b, size_t{9});
151 c = std::min(c, size_t{9});
152 d = std::min(d, size_t{9});
153 return static_cast<style_specificity>(b * 100 + c * 10 + d);
154 }

170[[nodiscard]] constexpr bool matches(
171 style_selector::const_iterator lhs_it_first,
172 style_selector::const_iterator lhs_it_last,
173 style_path::const_iterator rhs_it_first,
174 style_path::const_iterator rhs_it_last) noexcept
175{
176 if (std::distance(lhs_it_first, lhs_it_last) > std::distance(rhs_it_first, rhs_it_last)) {
177 return false;
178 }
179
180 while (lhs_it_first != lhs_it_last and rhs_it_first != rhs_it_last) {
181 --lhs_it_last;
182 --rhs_it_last;
183
184 if (not matches(*lhs_it_last, *rhs_it_last)) {
185 if (lhs_it_last->child_combinator or rhs_it_first == rhs_it_last) {
186 return false;
187 }
188 // Try to match the current selector segment again, but with the next segment in the path.
189 return matches(lhs_it_first, lhs_it_last + 1, rhs_it_first, rhs_it_last);
190 }
191 }
192
193 return lhs_it_last == lhs_it_first;
194}
