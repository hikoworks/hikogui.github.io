97 [[nodiscard]] layout::constraints make_constraints() const noexcept
98 {
99 auto r = layout::constraints{};
100
101 r.size = size_px;
102 r.margins = margins_px;
103 r.baseline_offset = [&] {
104 switch (vertical_alignment) {
105 case vertical_alignment::top:
106 return -x_height_px;
107 case vertical_alignment::middle:
108 return -x_height_px / 2.0f;
109 case vertical_alignment::bottom:
110 return 0.0f;
111 }
112 std::unreachable();
113 }();
114
115 r.baseline_priority = baseline_priority;
116 r.vertical_alignment = vertical_alignment;
117 return r;

129 [[nodiscard]] layout::constraints make_constraints_for_fixed_size() const noexcept
130 {
131 auto r = make_constraints();
132
133 r.baseline_offset = [&] {
134 auto const middle_px = height_px / 2.0f;
135 auto const x_middle_px = x_height_px / 2.0f;
136
137 switch (vertical_alignment) {
138 case vertical_alignment::top:
139 return -middle_px - x_middle_px
140 case vertical_alignment::middle:
141 return -x_middle_px;
142 case vertical_alignment::bottom:
143 return middle_px - x_middle_px;
144 }
145 std::unreachable();
146 }();
147
148 return r;
149 }

168 [[nodiscard]] extent2 concrete_size_px(extent2 natural_size, float scale) const noexcept
169 {
170 auto const concrete_width = [&] {
171 if (auto const *scalar_width = std::get_if<float>(&width)) {
172 return natural_size.width() * *scalar_width * scale;
173 } else {
174 return width_px;
175 }
176 }();
177
178 auto const concrete_height = [&] {
179 if (auto const *scalar_height = std::get_if<float>(&height)) {
180 return natural_size.height() * *scalar_height * scale;
181 } else {
182 return height_px;
183 }
184 }();
185
186 return {concrete_width, concrete_height};
187 }

204 [[nodiscard]] extent2 concrete_size_px(extent2 natural_size, float scale, extent2 layout_size) const noexcept
205 {
206 if (natural_size.width() == 0.0f or
207 natural_size.height() == 0.0f or
208 layout_size.width() == 0.0f or
209 layout_size.height() == 0.0f) {
210 // if the aspect-ratios can not be determined it is as-if object_fit::fill.
211 return layout_size;
212 }
213
214 auto const natural_aspect_ratio = natural_size.width() / natural_size.height();
215 auto const layout_aspect_ratio = layout_size.width() / layout_size.height();
216
217 auto const none_size = natural_size * scale;
218
219 auto const fill_size = layout_size;
220
221 auto const contain_size = natural_aspect_ratio < layout_aspect_ratio ?
222 extent2{layout_size.height() * natural_aspect_ratio, layout_size.height()} :
223 extent2{layout_size.width(), layout_size.width() / natural_aspect_ratio};
224
225 auto const cover_size = natural_aspect_ratio < layout_aspect_ratio ?
226 extent2{layout_size.width(), layout_size.width() / natural_aspect_ratio} :
227 extent2{layout_size.height() * natural_aspect_ratio, layout_size.height()};
228
229 auto const scale_down_size = none_size.width() < contain_size.width() ? none_size : contain_size;
230
231 switch (object_fit) {
232 case object_fit::none:
233 return none_size;
234 case object_fit::fill:
235 return fill_size;
236 case object_fit::contain:
237 return contain_size;
238 case object_fit::cover:
239 return cover_size;
240 case object_fit::scale_down:
241 return scale_down_size;
242 }
243 std::unreachable();
244 }

332 void set_pseudo_class(style_pseudo_class new_pseudo_class)
333 {
334 auto const t = trace<"style::set_pseudo_class">{};
335
336 assert(std::to_underlying(new_pseudo_class) < _loaded_properties.size());
337

350 [[nodiscard]] std::pair<style_path, properties_array_type const&> restyle(
351 unit::pixel_density density,
352 style_path const& parent_path,
353 properties_array_type const& parent_properties) noexcept
354 {
355 auto const t = trace<"style::restyle">{};
356
357 this->pixel_density = density;
358
359 auto path = parent_path;
360 path.emplace_back(_name, _id, _classes);
361
362 for (auto i = size_t{0}; i != style_pseudo_class_size; ++i) {
363 auto p = get_style_properties(path, static_cast<style_pseudo_class>(i));
364 p.apply(_override_properties);
365
366 _loaded_properties[i] = p * pixel_density;
367 _loaded_properties[i].inherit(parent_properties[i]);
368 }
369
370 for (auto i = size_t{0}; i != style_pseudo_class_size; ++i) {
371 auto const& src = _loaded_properties[i];
372 for (auto j = size_t{0}; j != style_pseudo_class_size; ++j) {
373 auto const& dst = _loaded_properties[j];
374 _pseudo_class_modifications[i + j * style_pseudo_class_size] = compare(src, dst);
375 }
376 }
377
378 update_properties(style_modify_mask::all);
379 _notifier(style_modify_mask::all, false);
380 return {path, _loaded_properties};
381 }

424 void update_properties(style_modify_mask mask)
425 {
426 auto const t = trace<"style::update_properties">{};
427
428 this->set_properties(_loaded_properties[std::to_underlying(_pseudo_class)], mask);
429
430 if (to_bool(mask & style_modify_mask::size)) {
431 if (auto const *pixel_width = std::get_if<unit::pixels_f>(&width)) {
432 width_px = pixel_width->in(unit::pixels);
433 } else {
434 assert(std::holds_alternative<float>(width));
435 width_px = 0.0f;
436 }
437
438 if (auto const *pixel_height = std::get_if<unit::pixels_f>(&height)) {
439 height_px = pixel_height->in(unit::pixels);
440 } else {
441 assert(std::holds_alternative<float>(height));
442 height_px = 0.0f;
443 }
444
445 size_px = extent2{width_px, height_px};
446 font_size_px = font_size.in(unit::pixels_per_em);
447
448 auto const &style = text_style.front();
449 auto const &primary_font = get_font(style.font_chain().front());
450
451 auto const scaled_font_size = font_size * style.scale();
452 x_height = round_as(unit::pixels, primary_font.metrics.x_height * scaled_font_size);
453 cap_height = round_as(unit::pixels, primary_font.metrics.cap_height * scaled_font_size);
454
455 x_height_px = x_height.in(unit::pixels);
456 cap_height_px = cap_height.in(unit::pixels);
457 }
458
459 if (to_bool(mask & style_modify_mask::margin)) {
460 margin_left_px = margin_left.in(unit::pixels);
461 margin_bottom_px = margin_bottom.in(unit::pixels);
462 margin_right_px = margin_right.in(unit::pixels);
463 margin_top_px = margin_top.in(unit::pixels);
464 padding_left_px = padding_left.in(unit::pixels);
465 padding_bottom_px = padding_bottom.in(unit::pixels);
466 padding_right_px = padding_right.in(unit::pixels);
467 padding_top_px = padding_top.in(unit::pixels);
468 margins_px = hi::margins{margin_left_px, margin_bottom_px, margin_right_px, margin_top_px};
469 horizontal_margins_px = hi::margins{margin_left_px, 0.0f, margin_right_px, 0.0f};
470 vertical_margins_px = hi::margins{0.0f, margin_bottom_px, 0.0f, margin_top_px};
471 padding_px = hi::margins{padding_left_px, padding_bottom_px, padding_right_px, padding_top_px};
472 horizontal_padding_px = hi::margins{padding_left_px, 0.0f, padding_right_px, 0.0f};
473 vertical_padding_px = hi::margins{0.0f, padding_bottom_px, 0.0f, padding_top_px};
474 }
475
476 if (to_bool(mask & style_modify_mask::weight)) {
477 border_width_px = border_width.in(unit::pixels);
478 border_bottom_left_radius_px = border_bottom_left_radius.in(unit::pixels);
479 border_bottom_right_radius_px = border_bottom_right_radius.in(unit::pixels);
480 border_top_left_radius_px = border_top_left_radius.in(unit::pixels);
481 border_top_right_radius_px = border_top_right_radius.in(unit::pixels);
482 border_radius_px = hi::corner_radii{
483 border_bottom_left_radius_px,
484 border_bottom_right_radius_px,
485 border_top_left_radius_px,
486 border_top_right_radius_px};
487 }
488
489 if (to_bool(mask & style_modify_mask::alignment)) {
490 alignment = hi::alignment{horizontal_alignment, vertical_alignment};
491 }
492 }
