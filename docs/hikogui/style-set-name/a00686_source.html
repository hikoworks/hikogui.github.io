21template<typename Value, typename Range>
22[[nodiscard]] constexpr Value get_first(Range &&range)
23{
24 auto it = std::ranges::begin(range);
25 auto last = std::ranges::end(range);
26
27 if (it == last) {
28 throw std::out_of_range{"Range is empty"};

63constexpr void clear_and_reserve(Container& container, size_t size)
64 requires requires(Container& c) {
65 c.clear();
66 c.capacity();
67 c.reserve(size);
68 }
69{
70 auto const capacity = container.capacity();
71 container.clear();
72
