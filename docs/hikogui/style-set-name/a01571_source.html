122 bool handle_event(gui_event const& event) noexcept override
123 {
124 switch (event.type()) {
125 case gui_event_type::mouse_move:
126 case gui_event_type::mouse_drag:
127 {
128 // Check the hover states of each button.
129 auto state_has_changed = false;
130 state_has_changed |= compare_store(hoverClose, closeRectangle.contains(event.mouse().position));
131 state_has_changed |= compare_store(hoverMinimize, minimizeRectangle.contains(event.mouse().position));
132 state_has_changed |= compare_store(hoverMaximize, maximizeRectangle.contains(event.mouse().position));
133 if (state_has_changed) {
134 request_redraw();
135 }
136 }
137 break;
138
139 case gui_event_type::mouse_exit:
140 hoverClose = false;
141 hoverMinimize = false;
142 hoverMaximize = false;
143 request_redraw();
144 return super::handle_event(event);
145
146 case gui_event_type::mouse_down:
147 if (event.mouse().cause.left_button) {
148 if (closeRectangle.contains(event.mouse().position)) {
149 pressedClose = true;
150
151 } else if (minimizeRectangle.contains(event.mouse().position)) {
152 pressedMinimize = true;
153
154 } else if (maximizeRectangle.contains(event.mouse().position)) {
155 pressedMaximize = true;
156 }
157 request_redraw();
158 return true;
159 }
160 break;
161
162 case gui_event_type::mouse_up:
163 if (event.mouse().cause.left_button) {
164 pressedClose = false;
165 pressedMinimize = false;
166 pressedMaximize = false;
167 request_redraw();
168
169 if (closeRectangle.contains(event.mouse().position)) {
170 return send_to_window({gui_event_type::window_close});
171
172 } else if (minimizeRectangle.contains(event.mouse().position)) {
173 return send_to_window({gui_event_type::window_minimize});
174
175 } else if (maximizeRectangle.contains(event.mouse().position)) {
176 switch (layout().window_size_state) {
177 case gui_window_size::normal:
178 return send_to_window({gui_event_type::window_maximize});
179
180 case gui_window_size::maximized:
181 return send_to_window({gui_event_type::window_normalize});
182
183 default:
184 hi_no_default();
185 }
186 }
187 return true;
188 }
189 break;
190
191 default:;
192 }
193 return super::handle_event(event);
194 }
