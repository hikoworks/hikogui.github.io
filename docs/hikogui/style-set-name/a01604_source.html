97 [[nodiscard]] box_constraints update_constraints() noexcept override
98 {
99 // Resolve as if in left-to-right mode, the grid will flip itself.
100 auto const resolved_alignment = resolve(style.alignment, true);
101
102 _grid.clear();
103 if (not _delegate->empty_icon(this) and not _delegate->empty_text(this)) {
104 // Both of the icon and text are set, so configure the grid to hold both.
105 if (resolved_alignment == horizontal_alignment::left) {
106 // icon text
107 _grid.add_cell(0, 0, _icon_widget.get());
108 _grid.add_cell(1, 0, _text_widget.get(), true);
109 } else if (resolved_alignment == horizontal_alignment::right) {
110 // text icon
111 _grid.add_cell(0, 0, _text_widget.get(), true);
112 _grid.add_cell(1, 0, _icon_widget.get());
113 } else if (resolved_alignment == vertical_alignment::top) {
114 // icon
115 // text
116 _grid.add_cell(0, 0, _icon_widget.get());
117 _grid.add_cell(0, 1, _text_widget.get(), true);
118 } else if (resolved_alignment == vertical_alignment::bottom) {
119 // text
120 // icon
121 _grid.add_cell(0, 0, _text_widget.get(), true);
122 _grid.add_cell(0, 1, _icon_widget.get());
123 } else {
124 // icon text (buttons want to be middle-center aligned).
125 _grid.add_cell(0, 0, _icon_widget.get());
126 _grid.add_cell(1, 0, _text_widget.get(), true);
127 }
128 } else if (not _delegate->empty_icon(this)) {
129 // Only the icon-widget is used.
130 _grid.add_cell(0, 0, _icon_widget.get());
131 } else if (not _delegate->empty_text(this)) {
132 // Only the text-widget is used.
133 _grid.add_cell(0, 0, _text_widget.get());
134 }
135
136 for (auto& cell : _grid) {
137 cell.set_constraints(cell.value->update_constraints());
138 }
139
140 return _grid.constraints(os_settings::left_to_right(), style.vertical_alignment);
141 }
