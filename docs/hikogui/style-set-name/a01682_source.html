139 void set_layout(layout::shape const& shape) noexcept override
140 {
141 _shape = shape;

143 auto const midline = shape.get_midline(style.x_height_px);
144
145 auto const horizontal_alignment = os_settings::alignment(style.horizontal_alignment);
146 _button_rectangle = align_to_middle(shape.rectangle(), style.size_px, horizontal_alignment, midline);
147
148 _check_glyph = find_glyph(elusive_icon::Ok);
149 auto const check_glyph_bb = _check_glyph.front_glyph_metrics().bounding_rectangle * style.font_size_px;
150 _check_glyph_rectangle = align(_button_rectangle, check_glyph_bb, alignment::middle_center());
151
152 _minus_glyph = find_glyph(elusive_icon::Minus);
153 auto const minus_glyph_bb = _minus_glyph.front_glyph_metrics().bounding_rectangle * style.font_size_px;
154 _minus_glyph_rectangle = align(_button_rectangle, minus_glyph_bb, alignment::middle_center());

157 void draw(draw_context const& context) const noexcept override
158 {
159 if (overlaps(context, _shape.rectangle)) {
160 context.draw_box(
161 _shape.translate_z() * _button_rectangle,
162 _shape.clipping_rectangle,
163 style.background_color,
164 style.border_color,
165 style.border_width_px,
166 border_side::inside,
167 style.border_radius_px);
168
169 switch (delegate->state(this)) {
170 case widget_value::on:
171 context.draw_glyph(
172 _shape.translate_z(0.1f) * _check_glyph_rectangle,
173 _shape.clipping_rectangle,
174 _check_glyph,
175 style.accent_color);
176 break;
177
178 case widget_value::off:
179 // Do nothing, the checkbox is off.
180 break;
181
182 default:
183 // Indeterminate
184 context.draw_glyph(
185 _shape.translate_z(0.1f) * _minus_glyph_rectangle,
186 _shape.clipping_rectangle,
187 _minus_glyph,
188 style.accent_color);
189 }
190 }
191
192 return super::draw(context);
193 }
