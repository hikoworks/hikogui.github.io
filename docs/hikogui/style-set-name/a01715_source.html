40 ~loop()
41 {
42 // Close all socket event handles.
43 while (_handles.size() > _socket_handle_idx) {
44 if (not WSACloseEvent(_handles.back())) {
45 hi_log_error("Could not clock socket event handle for socket {}. {}", _sockets.back(), get_last_error_message());
46 }
47
48 _handles.pop_back();
49 _sockets.pop_back();
50 }
51
52 if (_vsync_thread.joinable()) {
53 _vsync_thread.request_stop();
54 _vsync_thread.join();
55 }
56
57 if (not CloseHandle(_handles[_function_handle_idx])) {
58 hi_log_error("Could not close async-event handle. {}", get_last_error_message());
59 }
60 if (not CloseHandle(_handles[_vsync_handle_idx])) {
61 hi_log_error("Could not close vsync-event handle. {}", get_last_error_message());
62 }
63 if (not CloseHandle(_use_vsync_handle)) {
64 hi_log_error("Could not close use-vsync handle. {}", get_last_error_message());
65 }
66 }

68 loop() noexcept : _thread_id(current_thread_id())
69 {
70 // Create an level trigger event, to use as an on/off switch.
71 if (auto handle = CreateEventW(NULL, TRUE, TRUE, NULL)) {
72 _use_vsync_handle = handle;
73 } else {
74 hi_log_fatal("Could not create an use-vsync handle. {}", get_last_error_message());
75 }
76
77 // Create a pulse trigger event.
78 if (auto handle = CreateEventW(NULL, FALSE, FALSE, NULL)) {
79 _handles.push_back(handle);
80 _sockets.push_back(-1);
81 _socket_functions.emplace_back();
82 } else {
83 hi_log_fatal("Could not create an vsync-event handle. {}", get_last_error_message());
84 }
85
86 // Create a pulse trigger event.
87 if (auto handle = CreateEventW(NULL, FALSE, FALSE, NULL)) {
88 _handles.push_back(handle);
89 _sockets.push_back(-1);
90 _socket_functions.emplace_back();
91 } else {
92 hi_log_fatal("Could not create an async-event handle. {}", get_last_error_message());
93 }

326 int resume(std::stop_token stop_token = {}) noexcept
327 {
328 auto const is_main = this == _main.load(std::memory_order::relaxed);
329
330 // Microsoft recommends an event-loop that also renders to the screen to run at above normal priority.
331 auto const thread_handle = GetCurrentThread();
332
333 int original_thread_priority = GetThreadPriority(thread_handle);
334 if (original_thread_priority == THREAD_PRIORITY_ERROR_RETURN) {
335 original_thread_priority = THREAD_PRIORITY_NORMAL;
336 hi_log_error("GetThreadPriority() for loop failed {}", get_last_error_message());
337 }
338
339 if (is_main and original_thread_priority < THREAD_PRIORITY_ABOVE_NORMAL) {
340 if (not SetThreadPriority(thread_handle, THREAD_PRIORITY_ABOVE_NORMAL)) {
341 hi_log_error("SetThreadPriority() for loop failed {}", get_last_error_message());
342 }
343 }
344
345 _exit_code = {};
346 while (not _exit_code) {
347 resume_once(true);
348
349 if (stop_token.stop_possible()) {
350 if (stop_token.stop_requested()) {
351 // Stop immediately when stop is requested.
352 _exit_code = 0;
353 }
354 } else {
355 if (_render_functions.empty() and _function_fifo.empty() and _function_timer.empty() and
356 _handles.size() <= _socket_handle_idx) {
357 // If there is not stop token, then exit when there are no more resources to wait on.
358 _exit_code = 0;
359 }
360 }
361 }
362
363 // Set the thread priority back to what is was before resume().
364 if (is_main and original_thread_priority < THREAD_PRIORITY_ABOVE_NORMAL) {
365 if (not SetThreadPriority(thread_handle, original_thread_priority)) {
366 hi_log_error("SetThreadPriority() for loop failed {}", get_last_error_message());
367 }
368 }
369
370 return *_exit_code;
371 }

386 void resume_once(bool block = false) noexcept
387 {
388 using namespace std::chrono_literals;
389
390 hi_axiom(on_thread());
391
392 auto const is_main = this == _main.load(std::memory_order::relaxed);
393
394 auto current_time = std::chrono::utc_clock::now();
395 auto timeout = std::chrono::duration_cast<std::chrono::milliseconds>(_function_timer.current_deadline() - current_time);
396
397 timeout = std::clamp(timeout, 0ms, 100ms);
398 auto const timeout_ms = narrow_cast<DWORD>(timeout / 1ms);
399
400 // Only handle win32 messages when blocking.
401 // Since non-blocking is called from the win32 message-pump, we do not want to re-enter the loop.
402 auto const message_mask = is_main and block ? QS_ALLINPUT : 0;
403
404 auto const wait_r =
405 MsgWaitForMultipleObjects(narrow_cast<DWORD>(_handles.size()), _handles.data(), FALSE, timeout_ms, message_mask);
406
407 if (wait_r == WAIT_FAILED) {
408 hi_log_fatal("Failed on MsgWaitForMultipleObjects(), {}", get_last_error_message());
409
410 } else if (wait_r == WAIT_TIMEOUT) {
411 // handle_functions() and handle_timers() is called after every wake-up of MsgWaitForMultipleObjects
412 ;
413
414 } else if (wait_r == WAIT_OBJECT_0 + _vsync_handle_idx) {
415 // XXX Make sure this is not starving the win32 events.
416 // should we just empty the win32 events after every unblock?
417 handle_vsync();
418
419 } else if (wait_r == WAIT_OBJECT_0 + _function_handle_idx) {
420 // handle_functions() and handle_timers() is called after every
421 // wake-up of MsgWaitForMultipleObjects(). This is done below the
422 // if-cascade to make sure that the function-event is triggered
423 // after any kind of wake-up.
424 ;
425
426 } else if (wait_r >= WAIT_OBJECT_0 + _socket_handle_idx and wait_r < WAIT_OBJECT_0 + _handles.size()) {
427 auto const index = wait_r - WAIT_OBJECT_0;
428
429 WSANETWORKEVENTS events;
430 if (WSAEnumNetworkEvents(_sockets[index], _handles[index], &events) != 0) {
431 switch (WSAGetLastError()) {
432 case WSANOTINITIALISED:
433 hi_log_fatal("WSAStartup was not called.");
434 case WSAENETDOWN:
435 hi_log_fatal("The network subsystem has failed.");
436 case WSAEINVAL:
437 hi_log_fatal("One of the specified parameters was invalid.");
438 case WSAEINPROGRESS:
439 hi_log_warning(
440 "A blocking Windows Sockets 1.1 call is in progress, or the service provider is still processing a "
441 "callback "
442 "function.");
443 break;
444 case WSAEFAULT:
445 hi_log_fatal("The lpNetworkEvents parameter is not a valid part of the user address space.");
446 case WSAENOTSOCK:
447 // If somehow the socket was destroyed, lets just remove it.
448 hi_log_error("Error during WSAEnumNetworkEvents on socket {}: {}", _sockets[index], get_last_error_message());
449 _handles.erase(_handles.begin() + index);
450 _sockets.erase(_sockets.begin() + index);
451 _socket_functions.erase(_socket_functions.begin() + index);
452 break;
453 default:
454 hi_no_default();
455 }
456
457 } else {
458 // Because of how WSAEnumNetworkEvents() work we must only handle this specific socket.
459 _socket_functions[index](_sockets[index], socket_events_from_win32(events));
460 }
461
462 } else if (wait_r == WAIT_OBJECT_0 + _handles.size()) {
463 handle_gui_events();
464
465 } else if (wait_r >= WAIT_ABANDONED_0 and wait_r < WAIT_ABANDONED_0 + _handles.size()) {
466 auto const index = wait_r - WAIT_ABANDONED_0;
467 if (index == _vsync_handle_idx) {
468 hi_log_fatal("The vsync-handle has been abandoned.");
469
470 } else if (index == _function_handle_idx) {
471 hi_log_fatal("The async-handle has been abandoned.");
472
473 } else {
474 // Socket handle has been abandoned. Remove it from the handles.
475 hi_log_error("The socket-handle for socket {} has been abandoned.", _sockets[index]);
476 _handles.erase(_handles.begin() + index);
477 _sockets.erase(_sockets.begin() + index);
478 _socket_functions.erase(_socket_functions.begin() + index);
479 }
480
481 } else {
482 hi_no_default();
483 }
484
485 // Make sure timers are handled first, possibly they are time critical.
486 handle_timers();
487
488 // When functions are added wait-free, the function-event is never triggered.
489 // So handle messages after any kind of wake up.
490 handle_functions();
491
492 // The predicate functions are functions that are called when a certain
493 // condition is met.
494 handle_predicate_functions();
495 }

661 void handle_vsync() noexcept
662 {
663 // XXX Reduce the number of redraws for each window based on the refresh rate of the monitor they are located on.
664 // XXX handle maximum frame rate and update vsync thread
665 // XXX Update active windows more often than inactive windows.
666
667 if (not _vsync_thread.joinable()) {
668 // Fallback for the vsync_time advancing when the vsync thread is not running.
669 _vsync_time.store(std::chrono::utc_clock::now());
670 }
671
672 auto const display_time = _vsync_time.load(std::memory_order::relaxed) + std::chrono::milliseconds(30);
673
674 auto request_redraw = _request_redraw.exchange(false);
675 for (auto& render_function : _render_functions) {
676 if (auto rf = render_function.lock()) {
677 rf(display_time, request_redraw);
678 }
679 }
680
681 std::erase_if(_render_functions, [](auto& render_function) {
682 return render_function.expired();
683 });
684
685 if (_render_functions.empty()) {
686 // Stop the vsync thread when there are no more windows.
687 if (_vsync_thread.joinable()) {
688 _vsync_thread.request_stop();
689 }
690 }
691 }

737 void vsync_thread_update_dxgi_output() noexcept
738 {
739 if (not compare_store(_vsync_monitor_id, _selected_monitor_id.load(std::memory_order::relaxed))) {
740 return;
741 }
742
743 if (_vsync_monitor_output) {
744 _vsync_monitor_output->Release();
745 _vsync_monitor_output = nullptr;
746 }
747
748 IDXGIFactory *factory = nullptr;
749 if (FAILED(CreateDXGIFactory(__uuidof(IDXGIFactory), (void **)&factory))) {
750 hi_log_error_once("vsync:error:CreateDXGIFactory", "Could not IDXGIFactory. {}", get_last_error_message());
751 return;
752 }
753 hi_assert_not_null(factory);
754 auto d1 = defer([&] {
755 factory->Release();
756 });
757
758 IDXGIAdapter *adapter = nullptr;
759 if (FAILED(factory->EnumAdapters(0, &adapter))) {
760 hi_log_error_once("vsync:error:EnumAdapters", "Could not get IDXGIAdapter. {}", get_last_error_message());
761 return;
762 }
763 hi_assert_not_null(adapter);
764 auto d2 = defer([&] {
765 adapter->Release();
766 });
767
768 if (FAILED(adapter->EnumOutputs(0, &_vsync_monitor_output))) {
769 hi_log_error_once("vsync:error:EnumOutputs", "Could not get IDXGIOutput. {}", get_last_error_message());
770 return;
771 }
772
773 DXGI_OUTPUT_DESC description;
774 if (FAILED(_vsync_monitor_output->GetDesc(&description))) {
775 hi_log_error_once("vsync:error:GetDesc", "Could not get IDXGIOutput description. {}", get_last_error_message());
776 _vsync_monitor_output->Release();
777 _vsync_monitor_output = nullptr;
778 return;
779 }
780
781 if (description.Monitor != std::bit_cast<HMONITOR>(_vsync_monitor_id)) {
782 hi_log_error_once("vsync:error:not-primary-monitor", "DXGI primary monitor does not match desktop primary monitor");
783 _vsync_monitor_output->Release();
784 _vsync_monitor_output = nullptr;
785 return;
786 }
787
788 d2.cancel();
789 d1.cancel();
790 }

801 std::chrono::nanoseconds vsync_thread_update_time(bool on_sleep)
802 {
803 auto const ts = time_stamp_count(time_stamp_count::inplace_with_cpu_id{});
804 auto const new_time = time_stamp_utc::make(ts);
805
806 auto const was_sleeping = std::exchange(_vsync_time_from_sleep, on_sleep);
807 auto const old_time = _vsync_time.exchange(new_time, std::memory_order::acquire);
808
809 // If old_time was caused by sleeping it can not be used to calculate how long vsync was blocking.
810 return was_sleeping ? std::chrono::nanoseconds::max() : new_time - old_time;
811 }

813 void vsync_thread_wait_for_vblank() noexcept
814 {
815 using namespace std::chrono_literals;
816
817 vsync_thread_update_dxgi_output();
818
819 if (_vsync_monitor_output and FAILED(_vsync_monitor_output->WaitForVBlank())) {
820 hi_log_error_once("vsync:error:WaitForVBlank", "WaitForVBlank() failed. {}", get_last_error_message());
821 }
822
823 if (vsync_thread_update_time(false) < 1ms) {
824 hi_log_info_once("vsync:monitor-off", "WaitForVBlank() did not block; is the monitor turned off?");
825 Sleep(16);
826
827 // Fixup the time after the fallback sleep.
828 vsync_thread_update_time(true);
829 } else {
830 ++global_counter<"vsync:vertical-blank">;
831 }

861 void vsync_thread_proc(std::stop_token stop_token) noexcept
862 {
863 _vsync_thread_handle = GetCurrentThread();
864 set_thread_name("vsync");
865
866 while (not stop_token.stop_requested()) {
867 switch (WaitForSingleObject(_use_vsync_handle, 30)) {
868 case WAIT_TIMEOUT:
869 // When use_vsync is off wake the main loop every 30ms.
870 vsync_thread_update_time(true);
871
872 vsync_thread_update_priority(THREAD_PRIORITY_NORMAL);
873
874 ++global_counter<"vsync:low-priority">;
875 ++global_counter<"vsync:frame">;
876 SetEvent(_handles[_vsync_handle_idx]);
877 break;
878
879 case WAIT_OBJECT_0:
880 // When use_vsync is on wake the main loop based on the vertical-sync and pull_down.
881 vsync_thread_update_priority(THREAD_PRIORITY_TIME_CRITICAL);
882
883 vsync_thread_wait_for_vblank();
884

890 break;

895 break;

899 ResetEvent(_use_vsync_handle);

901 }
902 }
