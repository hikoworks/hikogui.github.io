15[[nodiscard]] inline uint16_t otype_class_definition_table_search_format1(std::span<std::byte const> bytes, glyph_id id)
16{
17 struct header_type {
18 big_uint16_buf_t format;
19 big_uint16_buf_t start_glyph_id;
20 big_uint16_buf_t glyph_count;
21 };
22
23 auto offset = size_t{0};
24 auto const header = implicit_cast<header_type>(offset, bytes);
25 assert(*header.format == 1);
26
27 auto const glyph_count = *header.glyph_count;
28 auto const start_glyph_id = *header.start_glyph_id;
29
30 if (id < start_glyph_id) {
31 return 0;
32 }
33
34 auto const index = id.value() - start_glyph_id;
35 if (index >= glyph_count) {
36 return 0;
37 }
38 auto const class_values = implicit_cast<big_uint16_buf_t>(offset, bytes, glyph_count);
39 return *class_values[index];
40}

42[[nodiscard]] inline uint16_t otype_class_definition_table_search_format2(std::span<std::byte const> bytes, glyph_id id)
43{
44 struct header_type {
45 big_uint16_buf_t format;
46 big_uint16_buf_t class_range_count;
47 };
48
49 struct class_range_type {
50 big_uint16_buf_t start_glyph_id;
51 big_uint16_buf_t end_glyph_id;
52 big_uint16_buf_t class_value;
53 };
54
55 auto offset = size_t{0};
56 auto const header = implicit_cast<header_type>(offset, bytes);
57 assert(*header.format == 2);
58
59 auto const class_range_count = *header.class_range_count;
60 auto const class_ranges = implicit_cast<class_range_type>(offset, bytes, class_range_count);
61
62 auto const it = std::lower_bound(class_ranges.begin(), class_ranges.end(), id, [](auto const& lhs, auto rhs) {
63 return *lhs.end_glyph_id < rhs;
64 });
65
66 if (it == class_ranges.end() or id < *it->start_glyph_id) {
67 return 0;
68 }
69 assert(id <= *it->end_glyph_id);
70
71 return *it->class_value;
72}

74[[nodiscard]] inline uint16_t otype_class_definition_table_search(std::span<std::byte const> bytes, glyph_id id)
75{
76 auto offset = size_t{0};
77 auto const format = implicit_cast<big_uint16_buf_t>(offset, bytes);
78
79 switch (*format) {
80 case 1:
81 return otype_class_definition_table_search_format1(bytes, id);
82 case 2:
83 return otype_class_definition_table_search_format2(bytes, id);
84 default:
85 throw otype_file_error("class definition table format not supported");
86 }
87}
