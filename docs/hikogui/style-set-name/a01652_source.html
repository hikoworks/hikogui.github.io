50template<axis Axis = axis::both>
51class scroll_widget : public widget {
52public:
53 using super = widget;
54 using horizontal_scroll_bar_type = scroll_bar_widget<axis::horizontal>;
55 using vertical_scroll_bar_type = scroll_bar_widget<axis::vertical>;

57 constexpr static hi::axis axis = Axis;

59 ~scroll_widget() {}

65 scroll_widget() noexcept : super()
66 {
67 hi_axiom(loop::main().on_thread());
68
69 _aperture = std::make_unique<scroll_aperture_widget>();
70 _aperture->set_parent(this);
71 _grid.add_cell(0, 0, _aperture.get());
72
73 if constexpr (std::to_underlying(axis & axis::horizontal)) {
74 _horizontal_scroll_bar = std::make_unique<horizontal_scroll_bar_type>(
75 _aperture->content_width, _aperture->aperture_width, _aperture->offset_x);
76 _horizontal_scroll_bar->set_parent(this);
77 _grid.add_cell(1, 0, _horizontal_scroll_bar.get());
78 }
79
80 if constexpr (std::to_underlying(axis & axis::vertical)) {
81 _vertical_scroll_bar = std::make_unique<vertical_scroll_bar_type>(
82 _aperture->content_height, _aperture->aperture_height, _aperture->offset_y);
83 _vertical_scroll_bar->set_parent(this);
84 _grid.add_cell(0, 1, _vertical_scroll_bar.get());
85 }
86
87 style.set_name("scroll-view");
88 }

117 [[nodiscard]] box_constraints update_constraints() noexcept override
118 {
119 for (auto& cell : _grid) {
120 cell.set_constraints(cell.value->update_constraints());
121 }
122 auto grid_constraints = _grid.constraints(os_settings::left_to_right(), style.vertical_alignment);

126 void set_layout(widget_layout const& context) noexcept override
127 {
128 super::set_layout(context);
129
130 _grid.set_layout(context.shape);
131
132 for (auto const& cell : _grid) {
133 auto shape = cell.shape;
134
135 if (cell.value == _aperture.get()) {
136 // This is the content. If the scroll bars are not visible make
137 // the shape of this cell overlap the entire layout.
138 auto x = shape.x();
139 auto y = shape.y();
140 auto width = shape.width();
141 auto height = shape.height();
142
143 if (not std::to_underlying(axis & axis::horizontal) or not _horizontal_scroll_bar->visible()) {
144 y = 0;
145 height = layout().height();
146 }
147 if (not std::to_underlying(axis & axis::vertical) or not _vertical_scroll_bar->visible()) {
148 x = 0;
149 width = layout().width();
150 }
151
152 shape.rectangle = aarectangle{x, y, width, height};
153 }
154
155 cell.value->set_layout(context.transform(shape, transform_command::level));
156 }
157 }

159 [[nodiscard]] hitbox hitbox_test(point2 position) const noexcept override
160 {
161 hi_axiom(loop::main().on_thread());
162
163 if (enabled()) {
164 auto r = _aperture->hitbox_test_from_parent(position);
165 if constexpr (std::to_underlying(axis & axis::horizontal)) {
166 r = _horizontal_scroll_bar->hitbox_test_from_parent(position, r);
167 }
168 if constexpr (std::to_underlying(axis & axis::vertical)) {
169 r = _vertical_scroll_bar->hitbox_test_from_parent(position, r);
170 }
171
172 if (layout().contains(position)) {
173 r = std::max(r, hitbox{id(), layout().elevation});
174 }
175 return r;
176
177 } else {
178 return {};
179 }
180 }
181 // @endprivatesection

186 std::unique_ptr<horizontal_scroll_bar_type> _horizontal_scroll_bar;
187 std::unique_ptr<vertical_scroll_bar_type> _vertical_scroll_bar;
188};
