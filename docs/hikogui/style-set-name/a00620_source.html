65 [[nodiscard]] constexpr T fold(R const& advances, T maximum_width, std::vector<size_t>& r) const
66 requires (std::convertible_to<std::ranges::range_value_t<R>, T>)
67 {
68 r.clear();
69
70 auto width = T{};
71 auto i = size_t{0};
72 while (i != advances.size()) {
73 auto const [next_i, line_width] = fit_line(advances, i, maximum_width);
74 assert(next_i > i);
75
76 r.push_back(next_i - i);
77
78 i = next_i;
79 width = std::max(width, line_width);
80 }
81
82 return width;
83 }

186 template<typename MatchFunc>
187 constexpr void LB_walk(MatchFunc match_func) noexcept
188 {
189 using enum unicode_line_break_class;
190
191 if (_infos.empty()) {
192 return;
193 }
194
195 auto cur = _infos.begin();
196 auto const last = _infos.end() - 1;
197 auto const last2 = _infos.end();
198 auto opportunity = _opportunities.begin() + 1;
199
200 auto cur_sp_class = XX;
201 auto cur_nu_class = XX;
202 auto prev_class = XX;
203 auto num_ri = 0_uz;
204 while (cur != last) {
205 auto const next = cur + 1;
206 auto const cur_class = unicode_line_break_class{*cur};
207 auto const next2_class = cur + 2 == last2 ? XX : unicode_line_break_class{*(cur + 2)};
208
209 // Keep track of classes followed by zero or more SP.
210 if (cur_class != SP) {
211 cur_sp_class = cur_class;
212 }
213
214 // Keep track of a "NU (NU|SY|IS)*" and "NU (NU|SY|IS)* (CL|CP)?".
215 if (cur_nu_class == CL) {
216 // Only a single CL|CP class may be at the end, then the number is closed.
217 cur_nu_class = XX;
218 } else if (cur_nu_class == NU) {
219 if (cur_class == CL or cur_class == CP) {
220 cur_nu_class = CL;
221 } else if (cur_class != NU and cur_class != SY and cur_class != IS) {
222 cur_nu_class = XX;
223 }
224 } else if (cur_class == NU) {
225 cur_nu_class = NU;
226 }
227
228 // Keep track of consecutive RI, but only count the actual RIs.
229 if (cur->original_class == RI) {
230 ++num_ri;
231 } else if (*cur != RI) {
232 num_ri = 0;
233 }
234
235 if (*opportunity == unicode_break_opportunity::unassigned) {
236 *opportunity = match_func(prev_class, cur, next, next2_class, cur_sp_class, cur_nu_class, num_ri);
237 }
238
239 prev_class = cur_class;
240 cur = next;
241 ++opportunity;
242 }
243 }

277 {
278 using enum unicode_line_break_class;
279 using enum unicode_break_opportunity;

282 return;
283 }
284
285 auto cur = _infos.begin();
286 auto const last = _infos.end() - 1;
287 auto opportunity = _opportunities.begin() + 1;
288
289 auto X = XX;
290 while (cur != last) {
291 auto const next = cur + 1;
292
293 if ((*cur == CM or *cur == ZWJ) and X != XX) {
294 // Treat all CM/ZWJ as X (if there is an X).
295 *cur |= X;

302 (*next == CM or *next == ZWJ)) {
303 // [^BK CR LF NL SP ZW] x [CM ZWJ]*
304 *opportunity = no;

432 template<typename T>
433 [[nodiscard]] constexpr T width(std::span<T const> advances, size_t first, size_t last) noexcept
434 {
435 auto const first_info = _infos.begin() + first;
436 auto const last_info = _infos.begin() + last;
437
438 // Find the position of the last visible character.
439 auto const last_visible_it = rfind_if_not(first_info, last_info, [](auto const& x) { return x.is_white_space(); });
440 if (last_visible_it == last_info) {
441 // None of the characters on this line are visible.
442 return T{};
443 }
444
445 // Calculate the width of the line, excluding the trailing whitespace.
446 auto const first_advance = advances.begin() + first;
447 auto const last_advance = advances.begin() + std::distance(first_info, last_visible_it) + 1;
448 return std::accumulate(first_advance, last_advance, T{}, [&](auto acc, auto x) {
449 return acc + x;
450 });
451 }

498 template<typename T, std::ranges::random_access_range R>
499 [[nodiscard]] constexpr std::pair<size_t, T> fit_line(R const& advances, size_t first, T maximum_width) const noexcept
500 requires (std::convertible_to<std::ranges::range_value_t<R>, T>)
501 {
502 using enum unicode_break_opportunity;
503
504 assert(_opportunities.size() == advances.size() + 1);
505 assert(_opportunities.back() == mandatory);
506
507 auto width = T{};
508 auto end_of_line = first;
509 auto width_of_line = T{};
510 auto found_visible = false;
511 for (auto i = first;; ++i) {
512 width += advances[i];
513 found_visible |= not _infos[i].is_white_space();
514
515 if (width > maximum_width and found_visible) {
516 if (i == first) {
517 // Not even a single character fits, return a single character.
518 return {first + 1, advances[first]};
519
520 } else if (end_of_line == first) {
521 // There is not a single word that fits in the maximum
522 // width. Return up to the previous character. trailing
523 // non-breaking whitespace will be included in this word
524 return {i, width - advances[i]};
525
526 } else {
527 // Return the previous word.
528 return {end_of_line, width_of_line};
529 }
530 }
531
532 // Check if there is a break opportunity after this character.
533 if (_opportunities[i + 1] == mandatory) {
534 // Found an end-of-line. If this character overflows the maximum width, it will be a white-space character.
535 return {i + 1, width};
536
537 } else if (_opportunities[i + 1] == yes) {
538 // Found a word, or a chunk of whitespace (possibly overflowing the maximum width).
539 end_of_line = i + 1;
540 width_of_line = width;
541 found_visible = false;
542 }
543 }
544
545 // We should never reach this point, as there is always a mandatory break at the end of the text.
546 std::unreachable();
547 }
