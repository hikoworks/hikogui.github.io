11inline void shaper::progress_to(state_type new_state)
12{
13 // Add prerequisites to new_state.
14 new_state.bidi_1 |= new_state.bidi_2;
15 new_state.fold |= new_state.bidi_2;
16 new_state.line_breaks |= new_state.bidi_2 or new_state.fold;
17 new_state.word_breaks |= new_state.runs;
18 new_state.base_code_points |= new_state.line_breaks or new_state.word_breaks or new_state.sentence_breaks or new_state.bidi_1;
19
20 if ((_state & new_state) == new_state) {
21 return;
22 }
23
24 if (new_state.base_code_points and not std::exchange(_state.base_code_points, true)) {
25 clear_and_resize(_base_code_points, _text.size(), U'\u0000');
26 for (auto i = size_t{0}; i != _text.size(); ++i) {
27 _base_code_points[i] = _text[i].starter();
28 }
29 }
30
31 if (new_state.line_breaks and not std::exchange(_state.line_breaks, true)) {
32 assert(_state.base_code_points);
33 _line_breaks.set_text(_base_code_points);
34 }
35
36 if (new_state.word_breaks and not std::exchange(_state.word_breaks, true)) {
37 assert(_state.base_code_points);
38 _word_breaks.set_text(_base_code_points);
39 }
40
41 if (new_state.sentence_breaks and not std::exchange(_state.sentence_breaks, true)) {
42 assert(_state.base_code_points);
43 _sentence_breaks.set_text(_base_code_points);
44 }
45
46 if (new_state.bidi_1 and not std::exchange(_state.bidi_1, true)) {
47 assert(_state.base_code_points);
48 _bidi.set_text(_base_code_points, _paragraph_direction);
49 }
50
51 if (new_state.runs and not std::exchange(_state.runs, true)) {
52 assert(_state.word_breaks);
53 shaper_make_run_lengths(_text, _word_breaks.opportunities(), _run_lengths);
54 shaper_make_run_ids(_run_lengths, _text.size(), _run_ids);
55 }
56
57 if (new_state.glyph_metrics and not std::exchange(_state.glyph_metrics, true)) {
58 assert(_state.runs);
59 shaper_collect_glyph_metrics(_text, _run_lengths, _style, _font_size, _glyphs, _metrics, _advances);
60 }
61
62 if (new_state.fold and not std::exchange(_state.fold, true)) {
63 assert(_state.line_breaks);
64 assert(_state.glyph_metrics);
65 _width_after_folding = _line_breaks.fold(_advances, _width, _line_lengths);
66 }
67
68 if (new_state.bidi_2 and not std::exchange(_state.bidi_2, true)) {
69 assert(_state.bidi_1);
70 assert(_state.fold);
71 _bidi.set_lines(_line_lengths);
72 }
73
74 if (new_state.substitutions and not std::exchange(_state.substitutions, true)) {}
75
76 if (new_state.positions and not std::exchange(_state.positions, true)) {}
77
78 assert((_state & new_state) == new_state);
79}
