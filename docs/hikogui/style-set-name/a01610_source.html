111 [[nodiscard]] box_constraints update_constraints() noexcept override
112 {
113 _content_constraints = _content->update_constraints();
114
115 // The margins of the content are internalized as padding inside the
116 // aperture. So that the scroll bars can be placed directly on the edge
117 // of the aperture.
118 _padding = max(_content_constraints.margins, style.padding_px);
119
120 // The minimum size of ther aperture is smaller than the content,
121 // otherwise the content would not be scrolled.
122 return {
123 style.size_px,
124 max(style.size_px, _content_constraints.preferred + _padding),
125 max(style.size_px, _content_constraints.maximum + _padding),
126 style.margins_px,
127 embed(_content_constraints.baseline, unit::pixels(_padding.bottom()), unit::pixels(_padding.top()))};
128 }
129
130 void set_layout(widget_layout const& context) noexcept override
131 {
132 super::set_layout(context);
133
134 auto const aperture_size = context.size() - _padding;
135 aperture_width = aperture_size.width();
136 aperture_height = aperture_size.height();
137
138 // Start scrolling with the preferred size as minimum, so
139 // that widgets in the content don't get unnecessarily squeezed.
140 content_width = std::min(*aperture_width, _content_constraints.preferred.width());
141 content_height = std::min(*aperture_height, _content_constraints.preferred.height());
142
143 // Make sure the offsets are limited to the scrollable area.
144 auto const offset_x_max = std::max(*content_width - *aperture_width, 0.0f);
145 auto const offset_y_max = std::max(*content_height - *aperture_height, 0.0f);
146 offset_x = std::clamp(*offset_x, 0.0f, offset_x_max);
147 offset_y = std::clamp(*offset_y, 0.0f, offset_y_max);
148
149 // The position of the content rectangle relative to the scroll view.
150 // The size is further adjusted if the either the horizontal or vertical scroll bar is invisible.
151 _content_shape = box_shape{
152 aarectangle{
153 -*offset_x + _padding.left(),
154 -*offset_y + _padding.bottom(),
155 *content_width,
156 *content_height},
157 lift(context.baseline(), unit::pixels(_padding.bottom()), unit::pixels(_padding.top()))};
158
159 // The content needs to be at a higher elevation, so that hitbox check
160 // will work correctly for handling scrolling with mouse wheel.
161 _content->set_layout(context.transform(_content_shape, transform_command::level, context.rectangle()));
162 }

164 [[nodiscard]] hitbox hitbox_test(point2 position) const noexcept override
165 {
166 hi_axiom(loop::main().on_thread());
167
168 auto r = _content->hitbox_test_from_parent(position);
169
170 if (layout().contains(position)) {
171 r = std::max(r, hitbox{id(), layout().elevation});
172 }
173 return r;

176 bool handle_event(gui_event const& event) noexcept override
177 {
178 hi_axiom(loop::main().on_thread());
179
180 if (event == gui_event_type::mouse_wheel) {
181 auto const new_offset_x =
182 *offset_x + std::round((unit::dips(event.mouse().wheel_delta.x()) * style.pixel_density).in(unit::pixels));
183 auto const new_offset_y =
184 *offset_y + std::round((unit::dips(event.mouse().wheel_delta.y()) * style.pixel_density).in(unit::pixels));
185 auto const max_offset_x = std::max(0.0f, *content_width - *aperture_width);
186 auto const max_offset_y = std::max(0.0f, *content_height - *aperture_height);
187
188 offset_x = std::clamp(new_offset_x, 0.0f, max_offset_x);
189 offset_y = std::clamp(new_offset_y, 0.0f, max_offset_y);
190 ++global_counter<"scroll_aperture_widget:mouse_wheel:relayout">;
191 request_relayout();
192 return true;
193 } else {
194 return super::handle_event(event);
195 }
196 }

198 void scroll_to_show(hi::aarectangle to_show) noexcept override
199 {
200 if (layout()) {
201 auto safe_rectangle = intersect(layout().rectangle(), layout().clipping_rectangle);
202 auto delta_x = 0.0f;
203 auto delta_y = 0.0f;
204
205 if (safe_rectangle.width() > _padding.width() and
206 safe_rectangle.height() > _padding.height()) {
207 // This will look visually better, if the selected widget is moved with some margin from
208 // the edge of the scroll widget. The margins of the content do not have anything to do
209 // with the margins that are needed here.
210 safe_rectangle = safe_rectangle - _padding;
211
212 if (to_show.right() > safe_rectangle.right()) {
213 delta_x = to_show.right() - safe_rectangle.right();
214 } else if (to_show.left() < safe_rectangle.left()) {
215 delta_x = to_show.left() - safe_rectangle.left();
216 }
217
218 if (to_show.top() > safe_rectangle.top()) {
219 delta_y = to_show.top() - safe_rectangle.top();
220 } else if (to_show.bottom() < safe_rectangle.bottom()) {
221 delta_y = to_show.bottom() - safe_rectangle.bottom();
222 }
223
224 // Scroll the widget
225 offset_x = std::round(offset_x + delta_x);
226 offset_y = std::round(offset_y + delta_y);
227 }
228
229 // There may be recursive scroll view, and they all need to move until the rectangle is visible.
230 if (auto* p = parent()) {
231 p->scroll_to_show(layout().to_parent * translate2(delta_x, delta_y) * to_show);
232 }
233
234 } else {
235 return super::scroll_to_show(to_show);
236 }
237 }
