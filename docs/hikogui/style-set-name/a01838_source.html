64 }
65 });

178 [[nodiscard]] virtual widget_phase phase() const noexcept
179 {
180 auto const pc = style.pseudo_class();
181 if ((pc & style_pseudo_class::phase_mask) == style_pseudo_class::disabled) {
182 return widget_phase::disabled;
183 } else if ((pc & style_pseudo_class::phase_mask) == style_pseudo_class::enabled) {
184 return widget_phase::enabled;
185 } else if ((pc & style_pseudo_class::phase_mask) == style_pseudo_class::hover) {
186 return widget_phase::hover;
187 } else if ((pc & style_pseudo_class::phase_mask) == style_pseudo_class::active) {
188 return widget_phase::active;
189 } else {
190 std::unreachable();
191 }
192 }
193
194 virtual void set_phase(widget_phase const& phase) noexcept
195 {
196 auto pc = style.pseudo_class();
197 pc &= ~style_pseudo_class::phase_mask;
198 pc |= [&] {
199 switch (phase) {
200 case widget_phase::disabled: return style_pseudo_class::disabled;
201 case widget_phase::enabled: return style_pseudo_class::enabled;
202 case widget_phase::hover: return style_pseudo_class::hover;
203 case widget_phase::active: return style_pseudo_class::active;
204 default: std::unreachable();
205 }
206 }();
207 style.set_pseudo_class(pc);
208 }

492inline widget_intf* get_if(widget_intf* start, widget_id id, bool include_invisible) noexcept
493{
494 hi_assert_not_null(start);
495
496 if (start->id() == id) {
497 return start;
498 }
499 for (auto& child : start->children(include_invisible)) {
500 if (auto const r = get_if(&child, id, include_invisible); r != nullptr) {
501 return r;
502 }
503 }
504 return nullptr;
505}
506
507inline widget_intf& get(widget_intf& start, widget_id id, bool include_invisible)
508{
509 if (auto r = get_if(std::addressof(start), id, include_invisible); r != nullptr) {
510 return *r;
511 }
512 throw not_found_error("get widget by id");
513}

515template<std::invocable<widget_intf&> Func>
516inline void apply(widget_intf& start, Func&& func, bool include_invisible = true)
517{
518 auto todo = std::vector<widget_intf*>{&start};
519
520 while (not todo.empty()) {
521 auto* tmp = todo.back();
522 todo.pop_back();
523
524 func(*tmp);
525
526 for (auto& child : tmp->children(include_invisible)) {
527 todo.push_back(&child);
528 }
529 }
