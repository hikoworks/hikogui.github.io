26template<std::ranges::input_range FeatureIndices>
27[[nodiscard]] inline generator<uint16_t> otype_feature_list_search_offsets(
28 std::span<std::byte const> bytes,
29 FeatureIndices&& feature_indices,
30 std::span<uint32_t> optional_feature_tags)
31{
32 auto feature_indices_it = std::ranges::begin(feature_indices);
33 auto const feature_indices_end = std::ranges::end(feature_indices);
34 assert(feature_indices_it != feature_indices_end);
35
36 struct feature_list_header_type {
37 big_uint16_buf_t feature_count;
38 };
39
40 struct feature_record_type {
41 uint8_t feature_tag[4];
42 big_uint16_buf_t feature_offset;
43 };
44
45 auto offset = size_t{0};
46 auto const feature_list_header = implicit_cast<feature_list_header_type>(offset, bytes);
47
48 auto const feature_records = implicit_cast<feature_record_type>(offset, bytes, *feature_list_header.feature_count);
49
50 // Add the required feature.
51 auto const required_feature_index = *feature_indices_it++;
52 if (required_feature_index != 0xffff and required_feature_index < feature_records.size()) {
53 co_yield *feature_records[required_feature_index].feature_offset;
54 }
55
56 // Add the optional selected features.
57 for (; feature_indices_it != feature_indices_end; ++feature_indices_it) {
58 auto const feature_index = *feature_indices_it;
59 if (feature_index >= feature_records.size()) {
60 throw otype_file_error("feature_index out of range");
61 }
62
63 auto const& feature_record = feature_records[feature_index];
64 for (auto const tag : optional_feature_tags) {
65 if (load<uint32_t, std::endian::big>(feature_record.feature_tag) == tag) {
66 co_yield *feature_record.feature_offset;
67 break;
68 }
69 }
70 }
71}

109template<std::ranges::input_range FeatureIndices>
110[[nodiscard]] inline lean_vector<uint16_t> otype_feature_table_search(
111 std::span<std::byte const> bytes,
112 FeatureIndices&& feature_indices,
113 std::span<uint32_t> optional_feature_tags)
114{
115 auto r = lean_vector<uint16_t>{};
116 for (auto const feature_table_offset :
117 otype_feature_list_search_offsets(bytes, std::forward<FeatureIndices>(feature_indices), optional_feature_tags)) {
118 auto const feature_table_bytes = bytes.subspan(gsl::narrow_cast<size_t>(feature_table_offset) * 2);
119
120 for (auto const lookup_index : otype_feature_table_lookup_indices(feature_table_bytes)) {
121 r.push_back(lookup_index);
122 }
123 }
124
125 // The list of lookup-indices should be sorted and unique.
126 // So that the lookup tables are used in the correct order on the glyphs.
127 std::sort(r.begin(), r.end());
128 r.erase(std::unique(r.begin(), r.end()), r.end());
129 return r;
130}
