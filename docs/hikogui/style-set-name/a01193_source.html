347 [[nodiscard]] friend constexpr aarectangle
348 align_to_middle(aarectangle haystack, extent2 needle, horizontal_alignment alignment, value_type middle) noexcept
349 {
350 assert(needle.width() <= haystack.width());
351 assert(needle.height() <= haystack.height());
352
353 auto const needle_half_height = needle.height() / value_type{2};
354 auto const needle_half_width = needle.width() / value_type{2};
355
356 auto const x = [&] {
357 switch (alignment) {
358 case horizontal_alignment::left:
359 case horizontal_alignment::flush:
360 case horizontal_alignment::justified:
361 return haystack.left();
362 case horizontal_alignment::right:
363 return haystack.right() - needle.width();
364 case horizontal_alignment::center:
365 return haystack.center() - needle_half_width;
366 }
367 std::unreachable();
368 }();
369
370 middle = std::clamp(
371 middle, haystack.bottom() + needle_half_height, haystack.top() - needle_half_height);
372
373 auto const y = middle - needle_half_height;
374 return {point2{x, y}, needle};
375 }

389 [[nodiscard]] friend constexpr bool overlaps(aarectangle const& lhs, aarectangle const& rhs) noexcept
390 {
391 if (lhs.empty() or rhs.empty()) {
392 return false;
393 }
394
395 auto const rhs_swap = rhs.v.zwxy();
396
397 // lhs.p0.x > rhs.p3.x | lhs.p0.y > rhs.p3.y
398 if (((lhs.v > rhs_swap).mask() & 0b0011) != 0) {
399 return false;
400 }
401
402 // lhs.p3.x < rhs.p0.x | lhs.p3.y < rhs.p0.y
403 if (((lhs.v < rhs_swap).mask() & 0b1100) != 0) {
404 return false;
405 }
406
407 return true;
408 }

551 [[nodiscard]] friend value_type distance(aarectangle const& lhs, point2 const& rhs) noexcept
552 {
553 auto const lhs_ = static_cast<array_type>(lhs);
554 auto const rhs_ = static_cast<array_type>(rhs);
555 // Only (x,y) of subsequent calculations are valid, (z,w) have garbage values.
556 auto const closest_point = max(min(rhs_, lhs_.zwzw()), lhs_);
557 auto const v_closest_point = closest_point - rhs_;
558 return hypot<0b0011>(v_closest_point).x();
559 }
