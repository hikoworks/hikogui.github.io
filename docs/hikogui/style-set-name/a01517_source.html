47color_primaries_to_RGBtoXYZ(float wx, float wy, float rx, float ry, float gx, float gy, float bx, float by) noexcept
48{
49 auto const w = vector3{wx, wy, 1.0f - wx - wy};
50 auto const r = vector3{rx, ry, 1.0f - rx - ry};
51 auto const g = vector3{gx, gy, 1.0f - gx - gy};
52 auto const b = vector3{bx, by, 1.0f - bx - by};
53
54 // Calculate white point's tristimulus values from coordinates
55 auto const W = vector3{1.0f * (w.x() / w.y()), 1.0f, 1.0f * (w.z() / w.y())};
56
57 // C is the chromaticity matrix.
58 auto const C = matrix3{r, g, b};
59
60 // solve tristimulus sums.
61 auto const S = scale3{~C * W};
62
63 return C * S;
64}
