151 [[nodiscard]] std::expected<bookmark, std::error_code> resolve(Suffixes&& suffixes) const
152 {
153 if (_path.empty()) {
154 return std::unexpected{std::make_error_code(std::errc::invalid_argument)};
155 }
156
157 // If the path is absolute then it is already resolved, only need to check
158 // if it actually exists. In the future we may want to look if the file
159 // has moved by tracking its inode.
160 if (_path.is_absolute()) {
161 if (std::filesystem::exists(_path)) {
162 return *this;
163 } else {
164 return std::unexpected{std::make_error_code(std::errc::no_such_file_or_directory)};
165 }
166 }
167
168 try {
169 return bookmark{get_path(_location, _path, std::forward<Suffixes>(suffixes)), _location};
170 } catch (std::system_error const& e) {
171 return std::unexpected{e.code()};
