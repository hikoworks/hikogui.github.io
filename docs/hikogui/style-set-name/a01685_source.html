61 void insert(
62 std::size_t first_column,
63 std::size_t first_row,
64 std::size_t last_column,
65 std::size_t last_row,
66 std::unique_ptr<widget> widget) noexcept
67 {
68 hi_axiom(loop::main().on_thread());
69 hi_axiom(first_column < last_column);
70 hi_axiom(first_row < last_row);
71
72 if (_grid.cell_in_use(first_column, first_row, last_column, last_row)) {
73 hi_log_fatal("cell ({},{}) of grid_widget is already in use", first_column, first_row);
74 }
75
76 widget->set_parent(this);
77 _grid.add_cell(first_column, first_row, last_column, last_row, std::move(widget));
78 hi_log_info("grid_widget::insert({}, {}, {}, {})", first_column, first_row, last_column, last_row);
79
80 ++global_counter<"grid_widget:insert:constrain">;
81 request_reconstrain();
82 }

112 void push_back(std::unique_ptr<widget> widget) noexcept
113 {
114 auto it = std::max_element(_grid.begin(), _grid.end(), [](auto const& a, auto const& b) {
115 return a.last_column < b.last_column;
116 });
117
118 if (it == _grid.end()) {
119 insert(0, 0, 1, 1, std::move(widget));
120 } else {
121 insert(it->last_column, 0, it->last_column + 1, 1, std::move(widget));
122 }
123 }
