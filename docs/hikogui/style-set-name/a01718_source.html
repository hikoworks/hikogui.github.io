30template<typename T = void, bool DestroyFrame = false>
31class task {
32public:
33 using value_type = T;
34 using notifier_type = notifier<void(value_type)>;
35 using callback_type = notifier_type::callback_type;
36
37 struct promise_type {
38 notifier_type notifier;
39 std::optional<value_type> value = {};
40 std::exception_ptr exception = nullptr;
41
42 template<std::convertible_to<value_type> NewValue>
43 void return_value(NewValue&& new_value) noexcept
44 {
45 value = std::forward<NewValue>(new_value);
46 }
47
48 void unhandled_exception() noexcept
49 {
50 exception = std::current_exception();
51 }
52
53 std::suspend_always final_suspend() noexcept
54 {
55 // The notifier is posted on the loop to ensure the coroutine
56 // gets completed. That way the callback will see the coroutine
57 // as completed.
58 if (value) {
59 // Trigger the notifier with the co_return value.
60 loop::local().post_function([this] {
61 this->notifier(*this->value);
62 });
63
64 } else {
65 // Notify also in case of exception.
66 hi_assert_not_null(exception);
67 if constexpr (std::is_default_constructible_v<value_type>) {
68 loop::local().post_function([this] {
69 this->notifier(value_type{});
70 });
71 }
72 }
73
74 return {};
75 }
76
77 task get_return_object() noexcept
78 {
79 return task{handle_type::from_promise(*this)};
80 }
81
84 std::suspend_never initial_suspend() noexcept
85 {
86 return {};
87 }
88
89 template<convertible_to_awaitable RHS>
90 decltype(auto) await_transform(RHS&& rhs)
91 {
92 return awaitable_cast<std::remove_cvref_t<RHS>>{}(std::forward<RHS>(rhs));
93 }
94 };
95
96 using handle_type = std::coroutine_handle<promise_type>;
97
98 task(handle_type coroutine) noexcept : _coroutine(coroutine) {}
99
100 ~task()
101 {
102 if (DestroyFrame and _coroutine) {
103 _coroutine.destroy();
104 }
105 }
106
107 task() = default;
108
109 // task can not be copied because it tracks if the co-routine must be destroyed by the
110 // shared_ptr to the value shared between task and the promise.
111 task(task const&) = delete;
112 task& operator=(task const&) = delete;
113
114 task(task&& other) noexcept
115 {
116 _coroutine = std::exchange(other._coroutine, {});
117 }
118
119 task& operator=(task&& other) noexcept
120 {
121 _coroutine = std::exchange(other._coroutine, {});
122 return *this;
123 }

127 [[nodiscard]] bool started() const noexcept
128 {
129 return _coroutine;
130 }
131
134 [[nodiscard]] bool running() const noexcept
135 {
136 return _coroutine and not _coroutine.done();
137 }
138
141 [[nodiscard]] bool done() const noexcept
142 {
143 return _coroutine and _coroutine.done();
144 }
145
151 [[nodiscard]] value_type const& value() const
152 {
153 hi_axiom(done());
154
155 auto const& promise = _coroutine.promise();
156 if (promise.value) {
157 return *promise.value;
158
159 } else {
160 hi_axiom(promise.exception);
161 std::rethrow_exception(promise.exception);
162 }
163 }
164
170 [[nodiscard]] value_type const& operator*() const
171 {

183 [[nodiscard]] callback<void(value_type)> subscribe(Func&& func, callback_flags flags = callback_flags::synchronous) noexcept
184 {
185 return _coroutine.promise().notifier.subscribe(std::forward<Func>(func), flags);
186 }

190 auto operator co_await() const noexcept
191 {
192 return _coroutine.promise().notifier.operator co_await();
193 }
194
195private:

197 handle_type _coroutine;
