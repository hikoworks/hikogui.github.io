144 [[nodiscard]] box_constraints update_constraints() noexcept override
145 {
146 hi_assert_not_null(delegate);
147 hi_assert_not_null(_error_label_widget);
148 hi_assert_not_null(_scroll_widget);
149
150 if (_text_widget->focus()) {
151 // Update the optional error value from the string conversion when the text-widget has keyboard focus.
152 _error_label = delegate->validate(this, *_text);
153
154 } else {
155 // When field is not focused, simply follow the observed_value.
156 revert(false);
157 }
158
159 _scroll_constraints = _scroll_widget->update_constraints();
160
161 auto const scroll_width = 100;
162 auto const box_size = extent2{
163 _scroll_constraints.margins.left() + scroll_width + _scroll_constraints.margins.right(),
164 _scroll_constraints.margins.top() + _scroll_constraints.preferred.height() + _scroll_constraints.margins.bottom()};
165
166 auto size = box_size;
167 auto margins = style.margins_px;
168 if (_error_label->empty()) {
169 _error_label_constraints = _error_label_widget->update_constraints();
170
171 } else {
172 _error_label_constraints = _error_label_widget->update_constraints();
173 inplace_max(size.width(), _error_label_constraints.preferred.width());
174 size.height() += _error_label_constraints.margins.top() + _error_label_constraints.preferred.height();
175 inplace_max(margins.left(), _error_label_constraints.margins.left());
176 inplace_max(margins.right(), _error_label_constraints.margins.right());
177 inplace_max(margins.bottom(), _error_label_constraints.margins.bottom());
178 }
179
180 return {
181 size,
182 size,
183 size,
184 margins,
185 embed(_scroll_constraints.baseline, style.padding_bottom, style.padding_top)};
186 }

306 void revert(bool force) noexcept
307 {
308 hi_assert_not_null(delegate);
309 _text = delegate->text(this);
310 _error_label = label{};
311 }
312 void commit(bool force) noexcept
313 {
314 hi_axiom(loop::main().on_thread());
315 hi_assert_not_null(delegate);
316
317 if (*continues or force) {
318 auto text = *_text;
319
320 if (delegate->validate(this, text).empty()) {
321 // text is valid.
322 delegate->set_text(this, text);
323 }
324
325 // After commit get the canonical text to display from the delegate.
326 _text = delegate->text(this);
327 _error_label = label{};
328 }

337
338 auto const line = line_segment(get<0>(outline), get<1>(outline));
339 context.draw_line(layout(), translate3{0.0f, 0.5f, 0.1f} * line, theme().border_width(), focus_color());
