81 _button_widget_cbt();
82
83 _style_cbt = style.subscribe([&](style_modify_mask mask, bool restyle) {
84 if (not restyle and not to_bool(mask & style_modify_mask::layout)) {
85 return;
86 }
87
88 // Resolve as if in left-to-right mode, the grid will flip itself.
89 auto const resolved_alignment = resolve(style.alignment, true);
90
91 _grid.clear();
92 if (resolved_alignment == horizontal_alignment::left) {
93 // button label
94 _grid.add_cell(0, 0, grid_cell_type::button);
95 _grid.add_cell(1, 0, grid_cell_type::label, true);
96 } else if (resolved_alignment == horizontal_alignment::right) {
97 // label button
98 _grid.add_cell(0, 0, grid_cell_type::label, true);
99 _grid.add_cell(1, 0, grid_cell_type::button);
100 } else if (resolved_alignment == vertical_alignment::top) {
101 // button
102 // label
103 _grid.add_cell(0, 0, grid_cell_type::button);
104 _grid.add_cell(0, 1, grid_cell_type::label, true);
105 } else if (resolved_alignment == vertical_alignment::bottom) {
106 // label
107 // button
108 _grid.add_cell(0, 0, grid_cell_type::label, true);
109 _grid.add_cell(0, 1, grid_cell_type::button);
110 } else {
111 hi_no_default("alignment is not allowed to be middle-center.");
112 }
113 });
114
115 style.set_name("with-label");
116 }
