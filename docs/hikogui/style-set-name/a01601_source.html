105 [[nodiscard]] box_constraints update_constraints() noexcept override
106 {
107 hi_assert_not_null(_content);
108 hi_assert_not_null(_toolbar);
109
110 _content_constraints = _content->update_constraints();
111 _toolbar_constraints = _toolbar->update_constraints();
112
113 auto r = box_constraints{};
114 r.minimum.width() = std::max(
115 _toolbar_constraints.margins.left() + _toolbar_constraints.minimum.width() + _toolbar_constraints.margins.right(),
116 _content_constraints.margins.left() + _content_constraints.minimum.width() + _content_constraints.margins.right());
117 r.preferred.width() = std::max(
118 _toolbar_constraints.margins.left() + _toolbar_constraints.preferred.width() + _toolbar_constraints.margins.right(),
119 _content_constraints.margins.left() + _content_constraints.preferred.width() + _content_constraints.margins.right());
120 r.maximum.width() = std::min(
121 _toolbar_constraints.margins.left() + _toolbar_constraints.maximum.width() + _toolbar_constraints.margins.right(),
122 _content_constraints.margins.left() + _content_constraints.maximum.width() + _content_constraints.margins.right());
123
124 // clang-format off
125 r.minimum.height() =
126 _toolbar_constraints.margins.top() +
127 _toolbar_constraints.preferred.height() +
128 std::max(_toolbar_constraints.margins.bottom(), _content_constraints.margins.top()) +
129 _content_constraints.minimum.height() +
130 _content_constraints.margins.bottom();
131 r.preferred.height() =
132 _toolbar_constraints.margins.top() +
133 _toolbar_constraints.preferred.height() +
134 std::max(_toolbar_constraints.margins.bottom(), _content_constraints.margins.top()) +
135 _content_constraints.preferred.height() +
136 _content_constraints.margins.bottom();
137 r.maximum.height() =
138 _toolbar_constraints.margins.top() +
139 _toolbar_constraints.preferred.height() +
140 std::max(_toolbar_constraints.margins.bottom(), _content_constraints.margins.top()) +
141 _content_constraints.maximum.height() +
142 _content_constraints.margins.bottom();
143 // clang-format on
144
145 // The operating system also has a minimum and maximum size, these sizes
146 // are more important than the calculated sizes.
147 inplace_max(r.minimum.width(), os_settings::minimum_window_width());
148 inplace_max(r.minimum.height(), os_settings::minimum_window_height());
149
150 inplace_clamp(r.maximum.width(), r.minimum.width(), os_settings::maximum_window_width());
151 inplace_clamp(r.maximum.height(), r.minimum.height(), os_settings::maximum_window_height());
152
153 inplace_clamp(r.preferred.width(), r.minimum.width(), r.maximum.width());
154 inplace_clamp(r.preferred.height(), r.minimum.height(), r.maximum.height());
155
156 _can_resize_width = r.minimum.width() != r.maximum.width();
157 _can_resize_height = r.minimum.height() != r.maximum.height();
158
159 return r;
160 }

162 void set_layout(widget_layout const& context) noexcept override
163 {
164 super::set_layout(context);
165
166 auto const toolbar_height = _toolbar_constraints.preferred.height();
167 auto const between_margin = std::max(_toolbar_constraints.margins.bottom(), _content_constraints.margins.top());
168
169 auto const toolbar_rectangle = aarectangle{
170 point2{_toolbar_constraints.margins.left(), context.height() - toolbar_height - _toolbar_constraints.margins.top()},
171 point2{
172 context.width() - _toolbar_constraints.margins.right(), context.height() - _toolbar_constraints.margins.top()}};
173 _toolbar_shape = box_shape{toolbar_rectangle};
174
175 auto const content_rectangle = aarectangle{
176 point2{_content_constraints.margins.left(), _content_constraints.margins.bottom()},
177 point2{context.width() - _content_constraints.margins.right(), toolbar_rectangle.bottom() - between_margin}};
178 _content_shape = box_shape{content_rectangle};
179
180 _toolbar->set_layout(context.transform(_toolbar_shape));
181 _content->set_layout(context.transform(_content_shape));
182 }

189 [[nodiscard]] hitbox hitbox_test(point2 position) const noexcept override
190 {
191 constexpr float BORDER_WIDTH = 10.0f;
192
193 hi_axiom(loop::main().on_thread());
194
195 auto r = _toolbar->hitbox_test_from_parent(position);
196 r = _content->hitbox_test_from_parent(position, r);
197
198 auto const is_on_l_edge = position.x() <= BORDER_WIDTH;
199 auto const is_on_r_edge = position.x() >= (layout().width() - BORDER_WIDTH);
200 auto const is_on_b_edge = position.y() <= BORDER_WIDTH;
201 auto const is_on_t_edge = position.y() >= (layout().height() - BORDER_WIDTH);
202
203 // Corner resize has always priority.
204 if (is_on_l_edge and is_on_b_edge) {
205 if (_can_resize_width and _can_resize_height) {
206 return {id(), layout().elevation, hitbox_type::bottom_left_resize_corner};
207 } else if (_can_resize_width) {
208 return {id(), layout().elevation, hitbox_type::left_resize_border};
209 } else if (_can_resize_height) {
210 return {id(), layout().elevation, hitbox_type::bottom_resize_border};
211 }
212 } else if (is_on_r_edge and is_on_b_edge) {
213 if (_can_resize_width and _can_resize_height) {
214 return {id(), layout().elevation, hitbox_type::bottom_right_resize_corner};
215 } else if (_can_resize_width) {
216 return {id(), layout().elevation, hitbox_type::right_resize_border};
217 } else if (_can_resize_height) {
218 return {id(), layout().elevation, hitbox_type::bottom_resize_border};
219 }
220 } else if (is_on_l_edge and is_on_t_edge) {
221 if (_can_resize_width and _can_resize_height) {
222 return {id(), layout().elevation, hitbox_type::top_left_resize_corner};
223 } else if (_can_resize_width) {
224 return {id(), layout().elevation, hitbox_type::left_resize_border};
225 } else if (_can_resize_height) {
226 return {id(), layout().elevation, hitbox_type::top_resize_border};
227 }
228 } else if (is_on_r_edge and is_on_t_edge) {
229 if (_can_resize_width and _can_resize_height) {
230 return {id(), layout().elevation, hitbox_type::top_right_resize_corner};
231 } else if (_can_resize_width) {
232 return {id(), layout().elevation, hitbox_type::right_resize_border};
233 } else if (_can_resize_height) {
234 return {id(), layout().elevation, hitbox_type::top_resize_border};
235 }
236 }
237
238 // Border resize only has priority if there is no scroll-bar in the way.
239 if (r.type != hitbox_type::scroll_bar) {
240 if (is_on_l_edge and _can_resize_width) {
241 return {id(), layout().elevation, hitbox_type::left_resize_border};
242 } else if (is_on_r_edge and _can_resize_width) {
243 return {id(), layout().elevation, hitbox_type::right_resize_border};
244 } else if (is_on_b_edge and _can_resize_height) {
245 return {id(), layout().elevation, hitbox_type::bottom_resize_border};
246 } else if (is_on_t_edge and _can_resize_height) {
247 return {id(), layout().elevation, hitbox_type::top_resize_border};
248 }
249 }
250
251 return r;
252 }

255 using enum gui_event_type;
256
257 switch (event.type()) {
