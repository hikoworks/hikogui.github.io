73 bool handle_event(gui_event const& event) noexcept override
74 {
75 hi_axiom(loop::main().on_thread());
76
77 switch (event.type()) {
78 case gui_event_type::keyboard_enter:
79 set_focus(true);
80 this->scroll_to_show();
81 return true;
82
83 case gui_event_type::keyboard_exit:
84 set_focus(false);
85 return true;
86
87 case gui_event_type::mouse_enter:
88 set_hover(true);
89 return true;
90
91 case gui_event_type::mouse_exit:
92 set_hover(false);
93 return true;
94
95 case gui_event_type::mouse_down:
96 if (enabled() and event.mouse().cause.left_button) {
97 set_active(true);
98 handle_event(gui_event_type::gui_activate_stay);
99 }
100 return true;
101
102 case gui_event_type::mouse_up:
103 if (enabled() and event.mouse().cause.left_button) {
104 set_active(false);
105 }
106 return true;
107
108 case gui_event_type::window_activate:
109 set_front(true);
110 // All widgets need the active value set.
111 return false;
112
113 case gui_event_type::window_deactivate:
114 set_front(false);
115 // All widgets need the active value unset.
116 return false;
117
118 case gui_event_type::gui_activate_stay:
119 send_to_window(gui_event_type::gui_activate);
120 if (accepts_keyboard_focus(keyboard_focus_group::menu)) {
121 // By going forward and backward we select the current parent,
122 // the widget that opened the menu-stack.
123 send_to_window(gui_event_type::gui_widget_next);
124 send_to_window(gui_event_type::gui_widget_prev);
125 }
126 return true;
127
128 case gui_event_type::gui_activate_next:
129 send_to_window(gui_event_type::gui_activate);
130 return send_to_window(gui_event_type::gui_widget_next);
131
132 case gui_event_type::gui_widget_next:
133 send_to_window(
134 gui_event::window_set_keyboard_target(id(), keyboard_focus_group::normal, keyboard_focus_direction::forward));
135 return true;
136
137 case gui_event_type::gui_widget_prev:
138 send_to_window(
139 gui_event::window_set_keyboard_target(id(), keyboard_focus_group::normal, keyboard_focus_direction::backward));
140 return true;
141
142 case gui_event_type::gui_menu_next:
143 if (enabled() and accepts_keyboard_focus(keyboard_focus_group::menu)) {
144 send_to_window(
145 gui_event::window_set_keyboard_target(id(), keyboard_focus_group::menu, keyboard_focus_direction::forward));
146 return true;
147 }
148 break;
149
150 case gui_event_type::gui_menu_prev:
151 if (enabled() and accepts_keyboard_focus(keyboard_focus_group::menu)) {
152 send_to_window(
153 gui_event::window_set_keyboard_target(id(), keyboard_focus_group::menu, keyboard_focus_direction::backward));
154 return true;
155 }
156 break;
157
158 case gui_event_type::gui_toolbar_next:
159 if (enabled() and accepts_keyboard_focus(keyboard_focus_group::toolbar)) {
160 send_to_window(
161 gui_event::window_set_keyboard_target(id(), keyboard_focus_group::toolbar, keyboard_focus_direction::forward));
162 return true;
163 }
164 break;
165
166 case gui_event_type::gui_toolbar_prev:
167 if (enabled() and accepts_keyboard_focus(keyboard_focus_group::toolbar)) {
168 send_to_window(
169 gui_event::window_set_keyboard_target(id(), keyboard_focus_group::toolbar, keyboard_focus_direction::backward));
170 return true;
171 }
172 break;
173
174 default:;
175 }

177 return false;

180 bool handle_event_recursive(
181 gui_event const& event,
182 std::vector<widget_id> const& reject_list = std::vector<widget_id>{}) noexcept override
183 {
184 hi_axiom(loop::main().on_thread());
185
186 auto handled = false;
187
188 for (auto& child : visible_children()) {
189 handled |= child.handle_event_recursive(event, reject_list);
190 }
191
192 if (!std::ranges::any_of(reject_list, [&](auto const& x) {
193 return x == id();
194 })) {
195 handled |= handle_event(event);
196 }
197
198 return handled;
199 }

201 [[nodiscard]] virtual widget_id find_next_widget(
202 widget_id current_keyboard_widget,
203 keyboard_focus_group group,
204 keyboard_focus_direction direction) const noexcept override
205 {
206 hi_axiom(loop::main().on_thread());
207
208 auto found = false;
209
210 if (current_keyboard_widget == 0 and accepts_keyboard_focus(group)) {
211 // If there was no current_keyboard_widget, then return this if it accepts focus.
212 return id();
213
214 } else if (current_keyboard_widget == id()) {
215 found = true;
216 }
217
218 auto children_ = std::vector<widget_intf const *>{};
219 for (auto& child : visible_children()) {
220 children_.push_back(std::addressof(child));
221 }
222
223 if (direction == keyboard_focus_direction::backward) {
224 std::reverse(begin(children_), end(children_));
225 }
226
227 for (auto *child : children_) {
228 hi_axiom_not_null(child);
229
230 if (found) {
231 // Find the first focus accepting widget.
232 if (auto tmp = child->find_next_widget({}, group, direction); tmp != 0) {
233 return tmp;
234 }
235
236 } else {
237 auto tmp = child->find_next_widget(current_keyboard_widget, group, direction);
238 if (tmp == current_keyboard_widget) {
239 // The current widget was found, but no next widget available in the child.
240 // Try the first widget that does accept keyboard focus.
241 found = true;
242
243 } else if (tmp != 0) {
244 // Return the next widget that was found in the child-widget.
245 return tmp;
246 }
247 }
248 }
249
250 if (found) {
251 // Either:
252 // 1. current_keyboard_widget was nullptr; this widget, nor its child widgets accept focus.
253 // 2. current_keyboard_wigget was this; none of the child widgets accept focus.
254 // 3. current_keyboard_widget is a child; none of the following widgets accept focus.
255 return current_keyboard_widget;
256 }
257
258 return {};
259 }
