144 [[nodiscard]] box_constraints update_constraints() noexcept override
145 {
146 hi_assert_not_null(_off_label_widget);
147 hi_assert_not_null(_current_label_widget);
148 hi_assert_not_null(_overlay_widget);
149
150 _off_label_constraints = _off_label_widget->update_constraints();
151 _current_label_constraints = _current_label_widget->update_constraints();
152 _overlay_constraints = _overlay_widget->update_constraints();
153
154 // Make it so that the scroll widget can scroll vertically.
155 // XXX: This is a hack, the scroll widget should be able to calculate its own constraints.
156 _scroll_widget->minimum->height() = style.height_px;
157
158 auto const chevron_size = extent2{style.width_px, 0.0f};
159
160 auto const overlay_minimum = extent2{_overlay_constraints.minimum.width(), 0.0f};
161 auto const overlay_preferred = extent2{_overlay_constraints.preferred.width(), 0.0f};
162 auto const overlay_maximum = extent2{_overlay_constraints.maximum.width(), 0.0f};
163
164 auto const content_minimum = max(_off_label_constraints.minimum, _current_label_constraints.minimum, overlay_minimum);
165 auto const content_preferred =
166 max(_off_label_constraints.preferred, _current_label_constraints.preferred, overlay_preferred);
167 auto const content_maximum = max(_off_label_constraints.maximum, _current_label_constraints.maximum, overlay_maximum);
168 _content_padding = max(_off_label_constraints.margins, _current_label_constraints.margins, style.padding_px);
169 auto const content_baseline = max(_off_label_constraints.baseline, _current_label_constraints.baseline);
170
171 auto r = box_constraints{};
172 r.minimum = content_minimum + chevron_size + _content_padding;
173 r.preferred = content_preferred + chevron_size + _content_padding;
174 r.maximum = content_maximum + chevron_size + _content_padding;
175 r.margins = style.margins_px;
176 r.baseline = embed(content_baseline, unit::pixels(_content_padding.bottom()), unit::pixels(_content_padding.top()));
177 hi_axiom(r.holds_invariant());
178 return r;
179 }

181 void set_layout(widget_layout const& context) noexcept override
182 {
183 super::set_layout(context);
184
185 _chevron_box_rectangle = [&] {
186 if (os_settings::left_to_right()) {
187 return aarectangle{0.0f, 0.0f, style.width_px, context.height()};
188 } else {
189 return aarectangle{point2{context.right() - style.width_px, 0.0f}, point2{context.right(), context.top()}};
190 }
191 }();
192
193 auto const content_rectangle = [&] {
194 if (os_settings::left_to_right()) {
195 return aarectangle{
196 point2{_chevron_box_rectangle.right() + _content_padding.left(), _content_padding.bottom()},
197 point2{context.right() - _content_padding.right(), context.top() - _content_padding.top()}};
198 } else {
199 return aarectangle{
200 point2{_content_padding.left(), _content_padding.bottom()},
201 point2{_chevron_box_rectangle.left() - _content_padding.right(), context.top() - _content_padding.top()}};
202 }
203 }();
204
205 auto const content_shape = box_shape{
206 content_rectangle,
207 lift(context.baseline(), unit::pixels(_content_padding.bottom()), unit::pixels(_content_padding.top()))};
208
209 _chevron_glyph = find_glyph(elusive_icon::ChevronUp);
210 auto const chevron_glyph_bbox = _chevron_glyph.front_glyph_metrics().bounding_rectangle * style.font_size_px;
211 _chevron_rectangle = align(_chevron_box_rectangle, chevron_glyph_bbox, alignment::middle_center());
212
213 // The overlay itself will make sure the overlay fits the window, so we give the preferred size and position
214 // from the point of view of the selection widget.
215 // The overlay should start on the same left edge as the selection box and the same width.
216 // The height of the overlay should be the maximum height, which will show all the options.
217 auto const overlay_width = std::clamp(
218 context.width() - style.width_px, _overlay_constraints.minimum.width(), _overlay_constraints.maximum.width());
219 auto const overlay_height = _overlay_constraints.preferred.height();
220 auto const overlay_x = os_settings::left_to_right() ? style.width_px : context.width() - style.width_px - overlay_width;
221 auto const overlay_y = std::round((context.height() - overlay_height) / 2.0f);
222 auto const overlay_rectangle_request = aarectangle{overlay_x, overlay_y, overlay_width, overlay_height};
223 auto const overlay_rectangle = make_overlay_rectangle(overlay_rectangle_request);
224 _overlay_widget->set_layout(context.transform(box_shape{overlay_rectangle}, transform_command::overlay));
225
226 _off_label_widget->set_layout(context.transform(content_shape));
227 _current_label_widget->set_layout(context.transform(content_shape));
228 }

243 bool handle_event(gui_event const& event) noexcept override
244 {
245 switch (event.type()) {
246 case gui_event_type::mouse_up:
247 if (enabled() and not delegate->empty(this) and layout().rectangle().contains(event.mouse().position)) {
248 return handle_event(gui_event_type::gui_activate);
249 }
250 return true;
251
252 case gui_event_type::gui_activate_next:
253 // Handle gui_active_next so that the next widget will NOT get keyboard focus.
254 // The previously selected item needs the get keyboard focus instead.
255 case gui_event_type::gui_activate:
256 if (enabled() and not delegate->empty(this)) {
257 switch (_overlay_state) {
258 case overlay_state_type::closed:
259 open_overlay();
260 break;
261 case overlay_state_type::open:
262 close_overlay();
263 break;
264 case overlay_state_type::closing:
265 break;
266 default:
267 std::unreachable();
268 }
269 }
270 ++global_counter<"selection_widget:gui_activate:relayout">;
271 request_relayout();
272 return true;
273
274 case gui_event_type::gui_cancel:
275 close_overlay();
276 return true;
277
278 default:;
279 }
280
281 return super::handle_event(event);
282 }

284 [[nodiscard]] hitbox hitbox_test(point2 position) const noexcept override
285 {
286 hi_axiom(loop::main().on_thread());
287
288 if (not enabled()) {
289 return {};
290 }
291
292 auto r = hitbox{};
293
294 if (_overlay_state == overlay_state_type::open) {
295 r = _overlay_widget->hitbox_test_from_parent(position);
296 }
297
298 if (layout().contains(position)) {
299 r = std::max(
300 r, hitbox{id(), layout().elevation, not delegate->empty(this) ? hitbox_type::button : hitbox_type::_default});
301 }
302
303 return r;

347 void open_overlay() noexcept
348 {
349 hi_axiom(loop::main().on_thread());
350
351 if (auto focus_id = delegate->keyboard_focus_id(this)) {
352 _overlay_state = overlay_state_type::open;
353 send_to_window(gui_event::window_set_keyboard_target(*focus_id, keyboard_focus_group::menu));
354 request_redraw_window();

358 void close_overlay() noexcept
359 {
360 hi_axiom(loop::main().on_thread());
361
362 if (_overlay_state == overlay_state_type::open) {
363 _overlay_state = overlay_state_type::closing;
364 _overlay_close_start = std::chrono::utc_clock::now();
365 request_redraw_window();
366 }
367 }

369 void force_close_overlay() noexcept
370 {
371 if (_overlay_state != overlay_state_type::closed) {
372 _overlay_state = overlay_state_type::closed;
373 request_redraw_window();
374 }
375 }

377 void animate_overlay(utc_nanoseconds display_time_point) noexcept
378 {
379 hi_axiom(loop::main().on_thread());
380
381 switch (_overlay_state) {
382 case overlay_state_type::open:
383 break;
384 case overlay_state_type::closing:
385 if (display_time_point >= _overlay_close_start + _overlay_close_delay) {
386 force_close_overlay();
387 } else {
388 request_redraw_window();
389 }
390 break;
391 case overlay_state_type::closed:
392 break;
393 default:
394 hi_no_default();
395 }
396 }
