17[[nodiscard]] inline aarectangle align(
18 aarectangle dst,
19 extent2 src,
20 hi::horizontal_alignment horizontal_alignment,
21 hi::vertical_alignment vertical_alignment,
22 unit::pixels_f cap_height)
23{
24 auto x = dst.left();
25 auto y = dst.bottom();
26 auto width = src.width();
27 auto height = src.height();
28
29 switch (horizontal_alignment) {
30 case hi::horizontal_alignment::left:
31 break;
32 case hi::horizontal_alignment::center:
33 x += (dst.width() - width) / 2.0f;
34 break;
35 case hi::horizontal_alignment::right:
36 x += dst.width() - width;
37 break;
38 default:
39 std::unreachable();
40 }
41
42 switch (vertical_alignment) {
43 case hi::vertical_alignment::top:
44 y += dst.height() - height / 2.0f;
45 y -= cap_height.in(unit::pixels) / 2.0f;
46 break;
47 case hi::vertical_alignment::middle:
48 y += (dst.height() - height) / 2.0f;
49 break;
50 case hi::vertical_alignment::bottom:
51 y -= height / 2.0f;
52 y += cap_height.in(unit::pixels) / 2.0f;
53 break;
54 default:
55 std::unreachable();
56 }
57
58 x = std::round(x);
59 y = std::round(y);
60 return aarectangle{x, y, width, height};
61}
