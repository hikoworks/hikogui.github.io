30inline void shaper_make_run_lengths(
31 std::span<grapheme const> text,
32 std::span<unicode_break_opportunity const> word_breaks,
33 std::vector<size_t>& r)
34{
35 r.clear();
36
37 // If the text is empty, then word_breaks is also empty.
38 if (text.empty()) {
39 return;
40 }
41
42 assert(text.size() + 1 == word_breaks.size());
43 // The last character is always a word break. Therfore we do not need to
44 // bound check the next grapheme access inside the loop.
45 assert(word_breaks.back() == unicode_break_opportunity::yes);
46
47 auto run_start = size_t{0};
48 for (auto i = size_t{0}; i != text.size(); ++i) {
49 // Is this is the last character of the run.
50 if (word_breaks[i + 1] == unicode_break_opportunity::yes) {
51 r.emplace_back(i + 1 - run_start);
52 run_start = i + 1;
53
54 } else if (text[i + 1].attributes() != text[i].attributes()) { // No bound check, see assert above.
55 r.emplace_back(i + 1 - run_start);
56 run_start = i + 1;
57 }
58 }
59}

70inline void shaper_make_run_ids(std::span<size_t const> run_lengths, size_t text_size, std::vector<size_t>& r)
71{
72 clear_and_resize(r, text_size, size_t{0});
73
74 auto id = 0;
75 auto r_it = r.begin();
76 for (auto length : run_lengths) {
77 for (auto i = size_t{0}; i != length; ++i, ++r_it) {
78 *r_it = id;
79 }
80 ++id;
81 }
82
83 assert(r_it == r.end());
84}

112inline void shaper_collect_glyph_metrics(
113 std::span<grapheme const> graphemes,
114 std::vector<size_t> const& run_lengths,
115 text_style_set const& style_set,
116 unit::pixels_per_em_f font_size,
117 std::vector<font_glyph_ids>& glyphs,
118 std::vector<shaper_grapheme_metrics>& metrics,
119 std::vector<unit::pixels_f>& advances)
120{
121 // A scratch pad to share the allocation of this vector between calls to
122 // find_glyphs() to improve the performance.
123 auto find_glyphs_scratch = std::vector<lean_vector<glyph_id>>{};
124
125 clear_and_resize(glyphs, graphemes.size(), font_glyph_ids{});
126 clear_and_resize(metrics, graphemes.size(), shaper_grapheme_metrics{});
127 clear_and_resize(advances, graphemes.size(), unit::pixels(0.0f));
128
129 auto attributes = grapheme_attributes{};
130 text_style style = text_style{};
131 auto style_font_size = font_size;
132 auto i = size_t{0};
133 for (auto run_length : run_lengths) {
134 assert(run_length != 0);
135
136 auto const run_of_graphemes = graphemes.subspan(i, run_length);
137 auto const run_of_glyphs = std::span{glyphs}.subspan(i, run_length);
138 auto const run_of_metrics = std::span{metrics}.subspan(i, run_length);
139 auto const run_of_advances = std::span{advances}.subspan(i, run_length);
140
141 if (style.empty() or run_of_graphemes.front().attributes() != attributes) {
142 attributes = run_of_graphemes.front().attributes();
143 style = style_set[attributes];
144 style_font_size = font_size * style.scale();
145 }
146
147 find_glyphs(run_of_graphemes, style.font_chain(), run_of_glyphs);
148
149 auto font_id = hi::font_id{};
150 hi::font* font = nullptr;
151 auto font_metrics = hi::font_metrics_px{};
152 for (auto j = size_t{0}; j != run_of_graphemes.size(); ++j) {
153 auto const& g = run_of_graphemes[j];
154 auto const& glyph_ids = run_of_glyphs[j];
155 auto& metrics = run_of_metrics[j];
156 auto& advance = run_of_advances[j];
157
158 if (font == nullptr or font_id != glyph_ids.font) {
159 font_id = glyph_ids.font;
160 font = std::addressof(get_font(font_id));
161 font_metrics = style_font_size * font->metrics;
162 }
163
164 metrics.glyphs = glyph_ids;
165 metrics.advance = style_font_size * font->get_advance(glyph_ids.front());
166 metrics.cap_height = font_metrics.cap_height;
167 metrics.ascender = font_metrics.ascender;
168 metrics.descender = font_metrics.descender;
169 metrics.line_gap = font_metrics.line_gap;
170 metrics.line_spacing = style.line_spacing();
171 metrics.paragraph_spacing = style.paragraph_spacing();
172 metrics.general_category = ucd_get_general_category(g.starter());
173 metrics.bracket_type = ucd_get_bidi_paired_bracket_type(g.starter());
174
175 // If the grapheme is a bracket, then find the mirrored glyph.
176 // Only the base glyph is mirrored, the combining glyphs are not.
177 // The mirrored glyph is used when the grapheme is displayed in RTL.
178 if (metrics.bracket_type != unicode_bidi_paired_bracket_type::n) {
179 auto const mirror_cp = ucd_get_bidi_mirroring_glyph(g.starter());
180 metrics.mirrored_glyph = font->find_glyph(mirror_cp);
181 }
182
183 advance = metrics.advance;
184 }
185
186 i += run_length;
187 }
188}

196inline void shaper_collect_line_metrics(
197 std::span<shaper_grapheme_metrics const> metrics,
198 std::span<size_t const> line_lengths,
199 std::vector<shaper_grapheme_metrics>& r)
200{
201 clear_and_resize(r, line_lengths.size(), shaper_grapheme_metrics{});
202
203 auto line_nr = size_t{0};
204 auto i_first = size_t{0};
205 for (auto const line_length : line_lengths) {
206 auto const i_last = i_first + line_length;
207
208 auto& line_metrics = r[line_nr];
209 for (auto i = i_first; i != i_last; ++i) {
210 auto const& grapheme_metrics = metrics[i];
211 line_metrics.cap_height = std::max(line_metrics.cap_height, grapheme_metrics.cap_height);
212 line_metrics.ascender = std::max(line_metrics.ascender, grapheme_metrics.ascender);
213 line_metrics.descender = std::max(line_metrics.descender, grapheme_metrics.descender);
214 line_metrics.line_gap = std::max(line_metrics.line_gap, grapheme_metrics.line_gap);
215 line_metrics.line_spacing = std::max(line_metrics.line_spacing, grapheme_metrics.line_spacing);
216 line_metrics.paragraph_spacing = std::max(line_metrics.paragraph_spacing, grapheme_metrics.paragraph_spacing);
217 }
218
219 ++line_nr;
220 i_first = i_last;
221 }
222}

257[[nodiscard]] inline std::vector<shaper_line_metrics> shaper_collect_line_metrics(
258 std::vector<shaper_grapheme_metrics> const& grapheme_metrics_range,
259 std::vector<size_t> const& line_lengths)
260{
261 auto r = std::vector<shaper_line_metrics>{};
262 r.reserve(line_lengths.size());
263
264 auto it = grapheme_metrics_range.begin();
265 for (auto const line_length : line_lengths) {
266 assert(line_length > 0);
267
268 auto const it_end = it + line_length;
269 auto const line_ends_paragraph = (it_end - 1)->general_category == unicode_general_category::Zp;
270
271 auto const it_visible = rfind_if(it, it_end, [](auto const& x) {
272 return is_visible(x.general_category);
273 });
274 auto const it_visible_end = it_visible != it_end ? it_visible + 1 : it_end;
275 auto const visible_line_span = std::span{it, it_visible_end};
276
277 auto metrics = shaper_line_metrics{};
278 for (auto const& g : visible_line_span) {
279 metrics.cap_height = std::max(metrics.cap_height, g.cap_height);
280 metrics.ascender = std::max(metrics.ascender, g.ascender);
281 metrics.descender = std::max(metrics.descender, g.descender);
282 metrics.line_gap = std::max(metrics.line_gap, g.line_gap);
283 metrics.spacing = std::max(metrics.spacing, line_ends_paragraph ? g.paragraph_spacing : g.line_spacing);
284 metrics.width += g.advance;
285 }
286
287 r.push_back(std::move(metrics));
288 it = it_end;
289 }
290
291 assert(not r.empty());
292 for (auto i = size_t{0}; i != r.size() - 1; ++i) {
293 r[i].advance = (r[i].descender + std::max(r[i].line_gap, r[i + 1].line_gap) + r[i + 1].ascender) * r[i].spacing;
294 }
295 r.back().advance = unit::pixels(0.0f);
296
297 return r;
298}

324[[nodiscard]] inline shaper_text_metrics
325shaper_collect_text_metrics(std::vector<shaper_line_metrics> const& line_metrics, hi::vertical_alignment alignment)
326{
327 auto r = shaper_text_metrics{};
328
329 if (line_metrics.empty()) {
330 return r;
331 }
332
333 auto const ascender = line_metrics.front().ascender;
334 auto const cap_height = line_metrics.front().ascender;
335 auto const descender = line_metrics.back().descender;
336
337 r.height = cap_height;
338 for (auto const& m : line_metrics) {
339 r.width = std::max(r.width, m.width);
340 r.height += m.advance;
341 }
342
343 // The overhang and underhang are used to determine the margins around the
344 // text.
345 if (ascender > cap_height) {
346 r.overhang = ascender - cap_height;
347 }
348 r.underhang = descender;
349
350 // The baseline of the middle line, or the average of the two middle lines.
351 // The distance from the bottom of the text to the baseline.
352 auto const middle_baseline_from_bottom = r.height - [&] {
353 auto const i = line_metrics.size() % 2 == 0 ? (line_metrics.size() / 2 - 1) : line_metrics.size() / 2;
354
355 auto const y =
356 std::accumulate(line_metrics.begin(), line_metrics.begin() + i + 1, unit::pixels(0.0f), [](auto sum, auto const& m) {
357 return sum + m.advance;
358 });
359
360 if (line_metrics.size() % 2 == 0) {
361 auto const y2 = y + line_metrics[i].advance;
362 return (y + y2) / 2.0f;
363 } else {
364 return y;
365 }
366 }();
367
368 // Calculate the offset to the middle baseline from the middle of the text.
369 auto const middle_baseline_from_middle = middle_baseline_from_bottom - r.height * 0.5f;
370
371 r.baseline_function = [&]() -> std::function<unit::pixels_f(unit::pixels_f)> {
372 switch (alignment) {
373 case vertical_alignment::top:
374 return [=](unit::pixels_f height) {
375 return height - cap_height;
376 };
377
378 case vertical_alignment::middle:
379 return [=](unit::pixels_f height) {
380 return height * 0.5f + middle_baseline_from_middle;
381 };
382
383 case vertical_alignment::bottom:
384 return [](unit::pixels_f height) {
385 return unit::pixels(0.0f);
386 };
387 }
388 std::unreachable();
389 }();
390
391 return r;
392}
