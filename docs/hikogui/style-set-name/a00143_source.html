45 template<typename F>
46 [[nodiscard]] constexpr constraints get_constraints(F const& f) noexcept
47 requires std::is_invocable_r_v<constraints, F>
48 {
49 _child_constraints = f();
50
51 auto const child_padding = max(_padding, _child_constraints.margins);
52
53 auto r = _child_constraints;
54 r.width += child_padding.horizontal();
55 r.height += child_padding.vertical();
56 r.margins = _margins;
57
58 switch (r.vertical_alignment) {
59 case vertical_alignment::top:
60 r.baseline_offset += -child_padding.top();
61 break;
62 case vertical_alignment::middle:
63 r.baseline_offset += (-child_padding.top() + child_padding.bottom()) * 0.5f;
64 break;
65 case vertical_alignment::bottom:
66 r.baseline_offset += child_padding.bottom();
67 break;
68 }
69
70 // The parent will likely have a higher priority than the child, when
71 // it is embedding the child with extra padding. This will make low
72 // priority labels in neighboring cells align with the baseline of
73 // a larger widget with a label in a different position.
74 if (_priority) {
75 r.baseline_priority = *_priority;
76 }
77 return r;
78 }

80 template<typename F>
81 constexpr void set_layout(layout::shape const& shape, F const& f) noexcept
82 requires std::is_invocable_r_v<void, F, shape>
83 {
84 auto child_shape = layout::shape{};
85 child_shape.rectangle = shape.rectangle - max(_padding, _child_constraints.margins);
86 child_shape.clipping_rectangle = intersect(shape.clipping_rectangle, child_shape.rectangle + _child_constraints.margins);
87 child_shape.elevation = shape.elevation + _elevation_offset;
88
89 child_shape.baseline = [&] {
90 if (shape.baseline_priority <= _child_constraints.baseline_priority) {
91 switch (_child_constraints._vertical_alignment) {
92 case vertical_alignment::top:
93 return child_shape.rectangle.top() + _child_constraints.baseline_offset;
94 case vertical_alignment::middle:
95 return child_shape.rectangle.middle() + _child_constraints.baseline_offset;
96 case vertical_alignment::bottom:
97 return child_shape.rectangle.bottom() + _child_constraints.baseline_offset;
98 }
99 }
100 }();
101
102 return f(child_shape);
103 }
