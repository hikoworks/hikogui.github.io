37template<axis Axis>
38class scroll_bar_widget : public widget {
39public:
40 using super = widget;
41
42 constexpr static hi::axis axis = Axis;
43
44 observer<float> offset;
45 observer<float> aperture;
46 observer<float> content;
47
48 template<forward_of<observer<float>> Content, forward_of<observer<float>> Aperture, forward_of<observer<float>> Offset>
49 scroll_bar_widget(
50 Content&& content,
51 Aperture&& aperture,
52 Offset&& offset) noexcept :
53 widget(),
54 content(std::forward<Content>(content)),
55 aperture(std::forward<Aperture>(aperture)),
56 offset(std::forward<Offset>(offset))
57 {
58 _content_cbt = this->content.subscribe([&](auto...) {
59 ++global_counter<"scroll_bar_widget:content:relayout">;
60 process_event({gui_event_type::window_relayout});
61 });
62 _aperture_cbt = this->aperture.subscribe([&](auto...) {
63 ++global_counter<"scroll_bar_widget:aperture:relayout">;
64 process_event({gui_event_type::window_relayout});
65 });
66 _offset_cbt = this->offset.subscribe([&](auto...) {
67 ++global_counter<"scroll_bar_widget:offset:relayout">;
68 process_event({gui_event_type::window_relayout});
69 });
70 }
71
72 ~scroll_bar_widget() {}

74 [[nodiscard]] box_constraints update_constraints() noexcept override
75 {
76 _layout = {};
77
78 if (mode() <= widget_mode::collapse) {
79 return {};
80 }
81
82 // The minimum size is twice the length of the slider, which is twice the theme().size()
83 if constexpr (axis == axis::vertical) {
84 return {
85 extent2{theme().icon_size(), theme().size() * 4},
86 extent2{theme().icon_size(), theme().size() * 4},
87 extent2{theme().icon_size(), large_number_v<int>}};
88 } else {
89 return {
90 extent2{theme().size() * 4, theme().icon_size()},
91 extent2{theme().size() * 4, theme().icon_size()},
92 extent2{large_number_v<int>, theme().icon_size()}};
93 }
94 }
95
96 void set_layout(widget_layout const& context) noexcept override
97 {
98 _layout = context;
99
100 if (mode() <= widget_mode::collapse) {
101 _slider_rectangle = {};
102 return;
103 }
104
105 // Calculate the position of the slider.
106 auto const slider_offset = std::round(*offset * travel_vs_hidden_content_ratio());
107 if constexpr (axis == axis::vertical) {
108 _slider_rectangle = aarectangle{0.0f, slider_offset, context.width(), slider_length()};
109 } else {
110 _slider_rectangle = aarectangle{slider_offset, 0.0f, slider_length(), context.height()};
111 }
112 }
113
114 [[nodiscard]] bool visible() const noexcept
115 {
116 return *aperture < *content;
117 }
118
119 void draw(draw_context const& context) noexcept override
120 {
121 if (mode() > widget_mode::invisible and overlaps(context, layout()) and visible()) {
122 draw_rails(context);
123 draw_slider(context);
124 }
125 }
126
127 hitbox hitbox_test(point2 position) const noexcept override
128 {
129 hi_axiom(loop::main().on_thread());
130
131 if (mode() >= widget_mode::partial and layout().contains(position) and visible() and
132 _slider_rectangle.contains(position)) {
133 return {id, _layout.elevation, hitbox_type::scroll_bar};
134 } else {
135 return {};
136 }
137 }
138
139 bool handle_event(gui_event const& event) noexcept override
140 {
141 switch (event.type()) {
142 case gui_event_type::mouse_down:
143 if (event.mouse().cause.left_button) {
144 // Record the original scroll-position before the drag starts.
145 _offset_before_drag = *offset;
146 return true;
147 }
148 break;
149
150 case gui_event_type::mouse_drag:
151 if (event.mouse().cause.left_button) {
152 // The distance the slider has to move relative to the slider position at the
153 // start of the drag.
154 auto const slider_movement = axis == axis::vertical ? event.drag_delta().y() : event.drag_delta().x();
155 auto const content_movement = round_cast<int>(slider_movement * hidden_content_vs_travel_ratio());
156 auto const new_offset = _offset_before_drag + content_movement;
157 offset = clamp_offset(new_offset);
158 return true;
159 }
160 break;
161
162 default:;
163 }
164
165 return super::handle_event(event);
166 }

168 [[nodiscard]] bool accepts_keyboard_focus(keyboard_focus_group group) const noexcept override
169 {
170 return false;
171 }
172
173 [[nodiscard]] color background_color() const noexcept override
174 {
175 return theme().fill_color(_layout.layer);
176 }
177
178 [[nodiscard]] color foreground_color() const noexcept override
179 {
180 if (phase() == widget_phase::hover) {
181 return theme().fill_color(_layout.layer + 2);
182 } else {
183 return theme().fill_color(_layout.layer + 1);
184 }
185 }

187private:
188 aarectangle _slider_rectangle;
189
190 float _offset_before_drag;
191
192 callback<void(float)> _content_cbt;
193 callback<void(float)> _aperture_cbt;
194 callback<void(float)> _offset_cbt;

200 [[nodiscard]] float clamp_offset(float new_offset) const noexcept
201 {
202 auto const scrollable_distance = std::max(0.0f, *content - *aperture);
203 return std::clamp(new_offset, 0.0f, scrollable_distance);
204 }
205
206 [[nodiscard]] float rail_length() const noexcept
207 {
208 hi_axiom(loop::main().on_thread());
209 return axis == axis::vertical ? layout().height() : layout().width();
210 }
211
212 [[nodiscard]] float slider_length() const noexcept
213 {
214 hi_axiom(loop::main().on_thread());
215
216 auto const preferred_length = [&] {
217 if (*content == 0.0f) {
218 return rail_length();
219 } else {
220 return std::round(*aperture * rail_length() / *content);
221 }
222 }();
223
224 return std::clamp(preferred_length, theme().size() * 2.0f, rail_length());
225 }
226
229 [[nodiscard]] float slider_travel_range() const noexcept
230 {
231 hi_axiom(loop::main().on_thread());
232 return rail_length() - slider_length();
233 }
234
237 [[nodiscard]] float hidden_content() const noexcept
238 {
239 hi_axiom(loop::main().on_thread());
240 return *content - *aperture;
241 }
242
247 [[nodiscard]] float hidden_content_vs_travel_ratio() const noexcept
248 {
249 hi_axiom(loop::main().on_thread());
250
251 auto const _slider_travel_range = slider_travel_range();
252 return _slider_travel_range != 0 ? std::round(hidden_content() / _slider_travel_range) : 0.0f;
253 }
254
259 [[nodiscard]] float travel_vs_hidden_content_ratio() const noexcept
260 {
261 hi_axiom(loop::main().on_thread());
262
263 auto const _hidden_content = hidden_content();
264 return _hidden_content != 0 ? slider_travel_range() / _hidden_content : 0.0f;
265 }
266
267 void draw_rails(draw_context const& context) noexcept
268 {
269 auto const corner_radii =
270 axis == axis::vertical ? hi::corner_radii{layout().width() * 0.5f} : hi::corner_radii{layout().height() * 0.5f};
271 context.draw_box(layout(), layout().rectangle(), background_color(), corner_radii);
272 }
273
274 void draw_slider(draw_context const& context) noexcept
275 {
276 auto const corner_radii = axis == axis::vertical ? hi::corner_radii{_slider_rectangle.width() / 2.0f} :
277 hi::corner_radii{_slider_rectangle.height() / 2.0f};
278
279 context.draw_box(layout(), translate_z(0.1f) * _slider_rectangle, foreground_color(), corner_radii);
280 }
281};
