236 if (previous_variant != variant()) {
237 switch (variant()) {
238 case gui_event_variant::mouse:

241 case gui_event_variant::grapheme:
242 _data = hi::grapheme{};
243 break;
244 case gui_event_variant::keyboard:
245 _data = keyboard_virtual_key{};

247 case gui_event_variant::keyboard_target:
248 _data = keyboard_target_data{};
249 break;
250 case gui_event_variant::rectangle:
251 _data = aarectangle{};
252 break;
253 case gui_event_variant::clipboard_data:
254 _data = gstring{};
255 default:;
256 }
